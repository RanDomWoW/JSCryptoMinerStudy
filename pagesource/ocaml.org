<span class='governing'>type</span> <span class='type'>tree</span> = Leaf <span class='keyword'>of</span> int | Node <span class='keyword'>of</span> tree * tree

<span class='governing'><span class='governing'>let rec</span></span> <span class='ocaml-function'>exists_leaf</span> <span class='ocaml-variable'>test tree </span>=
  <span class='keyword'>match</span> tree <span class='keyword'>with</span>
  | Leaf v -&gt; test v
  | Node (left, right) -&gt;
      exists_leaf test left
      || exists_leaf test right
