      function reorder_links(lead_element) {
         var elements = document.getElementsByClassName("most_list_link");
         
         lead_element.setAttribute("data-target-left", 0);
         
         var target = parseInt(window.getComputedStyle(lead_element).width, 10);
         
         for (var i = 0; i < elements.length; i++) {
            var element = elements[i];
            if (element !== lead_element) {
               element.setAttribute("data-target-left", target);
               target += parseInt(window.getComputedStyle(element).width, 10);
            }
         }
         
         for (var i = 0; i < elements.length; i++) {
            var element_opacity = elements[i] === lead_element ? "1" : "0.5";
            $(elements[i]).css("opacity", element_opacity).animate({left: elements[i].getAttribute("data-target-left")});
         }
      }

            if ($(this).attr("id") === link_element_id) {
               $(this).show();
            } else {
               $(this).hide();
            }
