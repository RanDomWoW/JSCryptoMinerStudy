        function initEyes() {
        var eyeOffsetNormal = 7;
        var eyeOffsetZoomed = 10;
        var adjust = 5; // hvor mye øyet skal flyttes opp og til venstre for å være sentrert.

        var eye1 = $('#eye1');
        var eye2 = $('#eye2');
        var eye1Cont = $('#eye1-cont');
        var eye2Cont = $('#eye2-cont');

        var isHovering = false;
        $(".trinn1-4").hover(function() {
            isHovering = true;
        }, function() {
            isHovering = false;
        }); 

        $(document).mousemove(function (e) {
            var eyeOffset = isHovering ? eyeOffsetZoomed : eyeOffsetNormal;
            
            var p1 = eye1.offset();
            var p2 = eye2.offset();

            var x = e.pageX;
            var y = e.pageY;

            dX1 = x - p1.left; 
            dY1 = y - p1.top; 

            dX2 = x - p2.left; 
            dY2 = y - p2.top; 

            var dist1 = Math.sqrt(dX1*dX1 + dY1*dY1);
            var dist2 = Math.sqrt(dX2*dX2 + dY2*dY2);

            var xxx1 = dist1 / eyeOffset;
            eye1Cont.css({'left': dX1 / xxx1 - adjust, 'top': dY1 / xxx1 - adjust});

            var xxx2 = dist2 / eyeOffset;
            eye2Cont.css({'left': dX2 / xxx2 - adjust, 'top': dY2 / xxx2 - adjust});
        });
    }

    $(document).ready(function () {
        if ($('body').hasClass('forsiden')) {
            initEyes();
        }
    });
