                            dataType: "json",
                            type: "GET",
                            contentType: "application/json; charset=utf-8",
                            dataFilter: function (data) { return data; },
                            success: function (data) {
                                var myData = JSON.parse(data.d);
                                var terms = myData.terms || [];
                                response($.map(terms,
                                    function (item) {
                                        return {
                                            label: item,
                                            value: item
                                        }
                                    }));
                            }
                        });
                    },
                    select: function (e, ui) {
                        var selectedval = ui.item.value;
                        $(".search-input").val(selectedval);
                        SearchBox($(".search-input").val());
                    }
                });


                $('#SearchContent').load(function () {
                    $('#loadingMessage').css('display', 'none');
                });
                $("#SearchContent").hide();
                $("div.searchResultsHeader").hide();
                $("#loadingMessage").hide();
            

                $(".searchbutton span").on('click', function () { SearchBox($(".searchbox").val()) });
                function SearchBox(searchval) {
                    var currentvalue = $.trim(searchval);
                    if ((currentvalue.length === 0)) {
                        $("#SearchContent").hide();
                        $("#primaryContent").show();
                        $("#SearchContent").html("");
                        $("div.searchResultsHeader").hide();
                        $("#loadingMessage").hide();

                    } else {
                        $("#loadingMessage").show();
                        $.ajax({
                            type: "GET",
                            dataType: 'html',
                            contentType: 'application/json; charset=utf-8',
                            url: '/Default2.aspx/Search',
                            data:
                            {
                                'query': JSON.stringify(currentvalue)
                            }
                        }).done(function (data) {
                            var result = JSON.parse(data);
                            $("#SearchContent").attr('src', result.d);
                            $("#content-primary").removeClass("inquiry-content2");
                            $("#SearchContent").show();
                            $("#primaryContent").hide();
                            $("div.searchResultsHeader").show();
                            $("div.searchResultsHeader").html("Search Results: " + '"' + currentvalue + '"');
                        }).fail(function (jqXHR, textStatus) { });
                    }
                };

                //In mobile mode, covers the entire screen with the menu popup window.
                var fullScreen = {
                    height: $(document).height(),
                    width: $(document).width()
                };

                $("div.RadMenuPopup").css(fullScreen);
            });

            function itemOpening(menu, args) {
                var item = args.get_item();
                var groupSettings = item._groupSettings;

                var attributes = item.get_attributes();
                var height = attributes.getAttribute("height");

                
                if (groupSettings && item.get_parent() !== menu) {
                    var expansionOffsetY = -height * item.get_index();
                    
                    groupSettings.set_offsetY(expansionOffsetY);
                }
