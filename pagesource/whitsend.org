var $j = jQuery.noConflict();
var _c = _h = _itemCount = 0;

window.onload=function(){
		initRotator();
}

function initRotator()
{
	getItemCount();
	addPlayOnClickEvent();
	$j("#movie_scroller ul li").hover(function(){clearInterval(_h)}, function(){play()});
	change(0);
	play();
}

function getItemCount()
{
	var play = $get('play');	
	_itemCount = play.children.length;	
}

function addPlayOnClickEvent()
{
	$j('#play a').click(function(){
		var i = $j(this).attr('alt') - 1;
		var altValue = $j(this).attr('alt');
		var lastIndex = _itemCount - 3;
		switch(altValue)
		{
			case 'Previous':				
				i = _c == 0 ? lastIndex : _c - 1;
				break;
			case 'Next':				
				i = _c == lastIndex ? 0 : _c + 1;
				break;
			default:
				i = altValue - 1;
				break;
		}
		clearInterval(_h);
		_c = i;
		play();
		change(i);        
	})
	
	$j('#PreviousID').mouseover(function(){	
		$get('PreviousImg').src='~/media/images/backgrounds/rotator_arrow_left_selected.ashx';				
	})
	
	$j('#PreviousID').mouseout(function(){				
		$get('PreviousImg').src='~/media/images/backgrounds/rotator_arrow_left.ashx';		
	})
	
	$j('#NextID').mouseover(function(){
		$get('NextImg').src='~/media/images/backgrounds/rotator_arrow_right_selected.ashx';		
	})
	
	$j('#NextID').mouseout(function(){			
		$get('NextImg').src='~/media/images/backgrounds/rotator_arrow_right.ashx';		
	})
}

function play()
{
    var duration = "7000";
    _h = setInterval("auto()", duration);
	setPalyPosition();
}
function change(i)
{
    $j('#play a').css('background-image','').eq(i+1).css('background-image','url(~/media/images/backgrounds/movie-scroller_selected.ashx)').blur();	
    $j("#movie_scroller ul li").hide().eq(i).fadeIn('slow');			
}
function auto()
{ 
	var lastItem = _itemCount - 4;
    _c = _c > lastItem ? 0 : _c + 1;
    change(_c);
}

function setPalyPosition()
{
	var play = $get('play');	
	var buttonSpace = 6;
	var playItemWidth = 22;
	var movieWidth = 382;
		
	var playWidth = playItemWidth * (_itemCount) + buttonSpace * (_itemCount - 1);
	var marginLeft = (movieWidth - playWidth)/2;
	play.style.margin = "-22px 0 0 "+marginLeft+"px";		
	play.style.display = 'inline';
}

function $get(id)
{
	return document.getElementById(id);
}
