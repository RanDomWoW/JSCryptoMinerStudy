function timeSince(seconds) {
    seconds = parseInt(seconds);
    if (seconds < 60) {
        return "Since " + seconds + " Seconds Ago";
    }
    var interval = parseInt(seconds / 31536000);
    if (interval > 0) {
        return "Since " + interval + " Years Ago";
    }
    interval = parseInt(seconds / 2592000);
    if (interval > 0) {
        return "Since " + interval + " Months Ago";
    }
    interval = parseInt(seconds / 86400);
    if (interval > 0) {
        return "Since " + interval + " Days Ago";
    }
    interval = parseInt(seconds / 3600);
    if (interval > 0) {
        return "Since " + interval + " Hours Ago";
    }
    interval = parseInt(seconds / 60);
    if (interval > 0) {
        return "Since " + interval + " Minutes Ago";
    }
    return "Since " + parseInt(seconds) + " Seconds Ago";
}
