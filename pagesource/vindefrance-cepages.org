function validSearch()
{
	if ($('search_id').value != "") {
		$('search').removeClassName("invalid");
		$('search').addClassName("valid");
	} else {
		$('search').removeClassName("valid");
		$('search').addClassName("invalid");
	}
	if ($('search').value.length > 0 && $('search_id').value != "") {
		sendSearch();
	}
}
function cleanSearch()
{
	$('search').value = "";
	$('search').focus();
}
function sendSearch()
{
	if ($('search_id').value != "") {
		
		if ($('search_id').value != "") {

		}	
		
		var url = "moteur-de-recherche-mets-vins?id_recipe=" + id_recipe;

	}
}
function createSearch(id, name)
{
