					} else sfCatID = "news"
				}
			}
		}
		if(typeof sfIsNews == "undefined") sfIsNews = false;

					if(sfIsNews) {
						sfPageID = parent.id.toLowerCase();
						sfPageID = sfCatID.toLowerCase()
					} else if(sfSelectedNav.hasClass("tab_link")){  // if this is a tab link, but not a news article

							isSecondaryTab = true
							// include the first tab ID in the page ID:

							sfSelectedNavSpanID = sfSelectedNavSpan.attr("id")

						} else sfPageID = parent.id.toLowerCase(); // is first tab link
					} else sfPageID = sfSelectedNavSpanID.toLowerCase();
				} else if(sfIsNews){

				sfCatID = sfPageID;
			}	
							
			if (typeof sfProdID == "undefined") sfProdID = ""

				bodyID = sfProdInitID
			} else if (sfProdID != "") {
				sfProdInitID = sfProdID
				bodyID = sfProdInitID
			} else if (sfProdID == "" && sfProdInitID != "") {
				bodyID = sfProdInitID
			} else if (sfProdID == "" && sfProdInitID == "") {
				bodyID = sfCatID
			}

			//window.console && console.log("prod2 = " + sfProdInitID)
