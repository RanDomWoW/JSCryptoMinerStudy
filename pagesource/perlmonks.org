          my ($cx,$cy) = split &#39;-&#39;,$_; 
          $cy &gt; $cx  ? 0 : ( $cx-$cy &lt; $minus + 1  ? 1 : 0)
        } 
        glob &#39;{&#39;.(join &#39;,&#39;,0..$x).&#39;}-&#39;.
