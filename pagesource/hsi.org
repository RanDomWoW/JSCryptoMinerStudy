                }
            },
            keys : {

            },
            iframe: {
              preload: true
           }
        });
    });

    function setParentUrl(element) {
        var receiver;

        if (typeof element.contentWindow !== "undefined") {
          receiver = element.contentWindow;
        } else {
          receiver = element.content[0].contentWindow;
        }

        try {
            receiver.postMessage(window.location.href, 'https://newsroom.humanesociety.org');
            console.log("Posted message to Newsroom iframe.");
        } catch(e) {
            console.log("Failed to post message to Newsroom iframe.");
        }
