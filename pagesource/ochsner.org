						scrollDirection('up');
					break;
					case 40: // down

					break;
					default: return; // exit this handler for other keys
				}

				var sections = $('.page-section');
				var index = 0;
				for(var x=0; x<sections.length; x++) {
					if ($(sections[x]).hasClass('active')) {
						if (direction === 'down') {
							index = x + 1;
						} else if (direction === 'up' && index > 0) {
							index = x - 1;
						} else {
							index = x - 1;
						}
					}
				}
				if (index > 0 && index < sections.length) {
					$('html, body').animate({
						scrollTop: $(sections[index]).position().top + navHeight
					}, 300, 'linear');
				} else if (index === 0 || index < sections.length) {
					$('html, body').animate({
						scrollTop: $(sections[index]).position().top
					}, 300, 'linear');
				}
			}
