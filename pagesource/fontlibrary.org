    const featuredTabs = document.querySelectorAll('#feature .tabs input');
    featuredTabs.forEach(function(tab) { tab.checked = false });
    
    function rotateFeatureTab() {
        const checkedTab = document.querySelector('#feature .tabs input:checked');
        if (checkedTab) {
            const nextTab = document.querySelector('#' + checkedTab.id + ' ~ input');
            if (nextTab) {
                checkedTab.checked = false;
                nextTab.checked = true;
            } else {
                featuredTabs[0].checked = true;
            }
        } else {
            featuredTabs[0].checked = true;
        }
    } 
    
    rotateFeatureTab();
    setInterval(rotateFeatureTab, 5000);
