		function submitContactRequest(contactType, organization, firstName, lastName, email, phone, phoneExt, reason) {
			var dataString = '{' + String.format("'contactType':'{0}', 'organization':'{1}', 'firstName':'{2}', 'lastName':'{3}', 'email':'{4}', 'phone':'{5}', 'phoneExt':'{6}', 'reason':'{7}', 'url':'{8}'",
					contactType,
					organization,
					firstName,
					lastName,
					email,
					phone,
					phoneExt,
					reason,
					window.location.href
				) + '}';
				$.ajax({
					type: "POST",
				url: "WebServices/GlobalWM.asmx/SubmitContactRequest",
					data: dataString,
					contentType: "application/json; charset=utf-8",
					dataType: "json",
				success: contactRequestSubmitted(contactType),
					error: function (xhr, ajaxOptions, thrownError) {
					console.log("Contact Request failed - " + xhr.status + " - " + thrownError);
					}
				});
			}

		function validateEmail(inputControl, validatorControl) {
			var email = $(inputControl).val();
			var isValid = email !== '' && emailRegex.test(email);

			if (isValid || email === '') { $(validatorControl).hide(); }
			else { $(validatorControl).show(); }
			return isValid;
		}
