  $(function () {
      var list = $('#partul');
      var cropper = $('#partul').parent();
      var list_item = list.children();
      var l = list_item.length;
      var buttons = $('.part_visual input');

      if (l < 9) {
          buttons.css('display', 'none');
          var percent_value = parseInt(100 / l);
          $('#partul').addClass('no_partners');
          list_item.css('width', percent_value + '%');
      } else {
          // 9 на страницу
          var keeper = true;
          list.append(list_item.clone());
          var cw = 0, step = 0, liw = 0, list_w = 0, current_pos = 0, max = 0;
          function set_size() {
              cw = cropper.width();
              step = cropper.width();
              liw = parseInt(cropper.width() / 9);
              step = liw;
              list_w = liw * l * 2;
              //max = -(list_w - cw);
              max = (list_item.length + 9) * liw - cw;
              list.children().css({'width': liw + 'px'});
              list.css({'width': list_w + 'px'});
          }
          set_size();

          function set_auto(dir) {
              if (keeper) {
                  keeper = false;
                  if (dir === 'next') {
                      current_pos -= step;
                      if (current_pos <= -max)
                          current_pos = -max;
                  } else if (dir === 'prev') {
                      if (current_pos === 0) {
                          list.css('margin-left', -max + 'px');
                          current_pos = -max;
                      }
                      current_pos += step;
                      if (current_pos > 0)
                          current_pos = 0;
                  } else {
                      return;
                  }
                  list.animate({'margin-left': current_pos + 'px'}, 250, function () {
                      if (-max === current_pos) {
                          list.css('margin-left', '0');
                          current_pos = 0;
                      }
                      keeper = true;
                  });
              }
          }

          buttons.click(function () {
              if (timeout)
                  clearTimeout(timeout);
              if (interval)
                  clearInterval(interval);
              this.blur();
              var dir = this.className.replace('reset ', '');
              set_auto(dir);
              timeout = setTimeout(function () {
                  interval = setInterval(function () {
                      set_auto('next');
                  }, 3000);
              }, 10000);
          });

          $(window).resize(set_size);

          var interval = setInterval(function () {
              set_auto('next');
          }, 3000);
          var timeout = null;
      }
      // list.append(list_item.clone());
  });
