    function loadDefaultFeed(itemsToReturn) {
        jQuery("#news").fadeOut(1, function () {
            jQuery(".loader").fadeIn(200, function () {

                var request = {};
                request.itemsToReturn = itemsToReturn;

                jQuery.ajax({
                    type: "POST",

                    dataType: "json",

                    data: JSON.stringify(request),
                    success: onGetFeed,
                    error: onNewsFeedError

    function loadFeed(feedId, itemsToReturn) {
        jQuery("#news").fadeOut(500, function () {
            jQuery(".loader").fadeIn(200, function () {

                var request = {};

                request.itemsToReturn = itemsToReturn;

                jQuery.ajax({
                    type: "POST",
                    contentType: "application/json; charset=utf-8",
                    dataType: "json",
                    url: serviceAddress + "/GetFeed",
                    data: JSON.stringify(request),

                    error: onNewsFeedError

            });
        });

    }

            jQuery("#news").fadeIn(200, function () {
                jQuery(".loader").hide();

        });
