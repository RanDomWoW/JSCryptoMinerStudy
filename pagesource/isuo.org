    function areaOver(areaId) {
        $('#areaLight').css('background-image', 'url(/images/map/map-gif/' + areaId + '.gif)');
        $('#link-area-' + areaId).css('background-color', '#ffb575');
        $('#bubble-' + areaId).SetBubblePopupInnerHtml($('#bubble-' + areaId).html());
        $('#bubble-' + areaId).ShowBubblePopup();
    }
    function areaOut(areaId) {
        $('#areaLight').css('background-image', 'none');

        $('#bubble-' + areaId).HideBubblePopup();
