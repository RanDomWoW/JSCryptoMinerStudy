				function syncPosition(el){
				    var current = this.currentItem;
				    $("#partner-logo")
				      	.find(".owl-item")

				function center(number){
				    var sync2visible = sync2Partner.data("owlCarousel").owl.visibleItems;
				    var num = number;
				    var found = false;
				    for(var i in sync2visible){
					    if(num === sync2visible[i]){
						   var found = true;
					    }
					}				 
				    if(found===false){
				      	if(num>sync2visible[sync2visible.length-1]){
				        	sync2Partner.trigger("owl.goTo", num - sync2visible.length+2)
				      }else{
				        	if(num - 1 === -1){
				          	num = 0;
				        }
				        sync2Partner.trigger("owl.goTo", num);
				      	}
				    } else if(num === sync2visible[sync2visible.length-1]){
				      	sync2Partner.trigger("owl.goTo", sync2visible[1])
				    } else if(num === sync2visible[0]){
				      	sync2Partner.trigger("owl.goTo", num-1)
