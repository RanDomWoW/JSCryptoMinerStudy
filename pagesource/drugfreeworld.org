function gcui_displayfact_init(b) {
    $(".gcui-displayfact-container-" + b + " .gcui-displayfact-item").hide();
    $(".gcui-displayfact-container-" + b + " .gcui-displayfact-item:first").addClass("active").show();
    var a = $(".gcui-displayfact-container-" + b);
    a.hover(function() {
        gcui_displayfact_pause_auto(b);
        $(".gcui-displayfact-container-" + b + " .gcui-displayfact-controls").fadeIn(250)
    }, function() {
        gcui_displayfact_unpause_auto(b);
        $(".gcui-displayfact-container-" + b + " .gcui-displayfact-controls").fadeOut(250)
    });
    $(".gcui-displayfact-container-" + b + " .prev").click(function() {
        gcui_displayfact_prev(b, true)
    });
    $(".gcui-displayfact-container-" + b + " .next").click(function() {
        gcui_displayfact_next(b, true)
    })
}

function gcui_displayfact_next(d, a) {
    var c = $(".gcui-displayfact-container-" + d + " .active");
    var b = $(".gcui-displayfact-container-" + d + " .active + .gcui-displayfact-item");
    if (!b.length) {
        b = $(".gcui-displayfact-container-" + d + " .gcui-displayfact-item:first")
    }
    _gcui_displayfact_swap(d, c, b, a)
}

function gcui_displayfact_prev(d, a) {
    var b = $(".gcui-displayfact-container-" + d + " .active");
    var c = $(".gcui-displayfact-container-" + d + " .active").prev();
    if (!(c && c.hasClass("gcui-displayfact-item"))) {
        c = $(".gcui-displayfact-container-" + d + " .gcui-displayfact-item:last")
    }
    _gcui_displayfact_swap(d, b, c, a)
}

function _gcui_displayfact_swap(d, c, b, a) {
    $(".gcui-displayfact-container-" + d + " .gcui-displayfact-item").removeClass("active").stop().dequeue();
    b.addClass("active");
    if (a) {
        c.hide();
        b.show()
    } else {
        c.fadeOut(300, function() {
            b.fadeIn(300)
        })
    }
}

function gcui_displayfact_set_auto(c) {
    var b = 5000;
    var a = setInterval(function() {
        gcui_displayfact_next(c)
    }, b);
    $(".gcui-displayfact-container-" + c).data("gcui_displayfact_intervalid", a);
    $(".gcui-displayfact-container-" + c).data("gcui_displayfact_intervalid_saved", null)
}

function gcui_displayfact_unset_auto(b) {
    var a = $(".gcui-displayfact-container-" + b).data("gcui_displayfact_intervalid");
    if (a) {
        clearInterval(a)
    }
    $(".gcui-displayfact-container-" + b).data("gcui_displayfact_intervalid", null);
    $(".gcui-displayfact-container-" + b).data("gcui_displayfact_intervalid_saved", null)
}

function gcui_displayfact_pause_auto(b) {
    var a = $(".gcui-displayfact-container-" + b).data("gcui_displayfact_intervalid");
    if (a) {
        clearInterval(a);
        $(".gcui-displayfact-container-" + b).data("gcui_displayfact_intervalid_saved", a);
        $(".gcui-displayfact-container-" + b).data("gcui_displayfact_intervalid", null)
    }
}

function gcui_displayfact_unpause_auto(a) {
    if ($(".gcui-displayfact-container-" + a).data("gcui_displayfact_intervalid_saved")) {
        gcui_displayfact_set_auto(a)
    }
}
