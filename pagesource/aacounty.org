<script>
var searchVal = "";
var url = "";
var buttonSwitch = "1";
var vSection = "all";
var featuredArr = [];

  function buildFeaturedArr() {
	var title = "";
	var keywords = "";
	var desc = "";
	var link = "";
	var external = "";
  
	  $.ajax({
		url: "http://www.aacounty.org/search-apps/featured_results?format=jsonp&sort=_title-asc",
			cache:false,
			dataType: "jsonp"}) 
			.done(function(results){ 
				for (i=0; i < results.CONTENTS.length; i++){
					title = results.CONTENTS[i]._title.VALUE; 
					keywords = results.CONTENTS[i].Keywords.VALUE; 
					desc = results.CONTENTS[i].Description.VALUE; 
					link = results.CONTENTS[i].Link.VALUE.LINK; 
					external = results.CONTENTS[i].Link.VALUE.LINKTYPE; 
					featuredArr.push({ "title" : title, "keywords" : keywords, "desc" : desc, "link" : link, "external" : external });
				}

//      all done 
        })
         .fail(function() { 
         //Handle failure 
         console.log("error");
		});
  
  }

function moveCursorToEnd(input) {
    var originalValue = input.val();
    input.val('');
    input.blur().focus().val(originalValue);
}
  
function uncheckAll(){
	//	uncheck all
	$("input:checkbox[name='dept']").each(function(i) {
		this.checked = false;
	});
}

$(document).ready(function(){
    //$("#searchBox").focus();
    
    var offset = $("#welcome").offset();
    
    if (window.screen.availWidth > 768){
        $("#searchModal").offset({
            top:offset.top
        });
    }
    
	buildFeaturedArr();
	
	
	if (window.screen.availWidth <= 560){
		mobileDisplay = true;
    }

    // When the user clicks anywhere outside of the modal, close it
	$("body").click(function(){
		$("#searchModal").fadeOut().removeClass("active");
        $("#results").html("");
        $("#footerDiv").html("");
		searchVal = "";
        uncheckAll();
		$("#InputSearch").val("");
		$("#searchBox").val("");
	});

	// Prevent events from getting pass .popup
	$("#searchModal").click(function(e){
	  e.stopPropagation();
	});

	
	$("#close").click(function(){
		searchVal = "";
        uncheckAll();
		$("#InputSearch").val("");
		$("#searchBox").val("");
		searchModal.style.display = "none";
    }); 

	$( "#toggle-entiresite" ).click(function(e){
        $( "#loading" ).show();
        buttonSwitch = "1";
        uncheckAll();
        $('#toggle-entiresite').prop('checked', true);         
		url = "http://www.aacounty.org/search-apps/site-search?search=";
        $( "#InputSearch" ).autocomplete("search");
	});
	
	$( "#toggle-services" ).click(function(e){ 
        $( "#loading" ).show();
		buttonSwitch = "2";

        $('#toggle-forms-publications').prop('checked', true);

	});
		
	$("#searchBox").keyup(function(){
        $('#toggle-entiresite').prop('checked', true);     
        url = "http://www.aacounty.org/search-apps/site-search?search=";

		$("#InputSearch").val(this.value);
		$("#InputSearch").focus();
		moveCursorToEnd($("#InputSearch"));

        if($("#InputSearch").val().length > 0){
            //$( "#results" ).html("");
            $( "#footerDiv" ).html("");
            $( "#loading" ).show();
        }
        else{
            $( "#loading" ).hide();
            $( "#results" ).html("<h4>0 results found</h4>");
        }
    });

    $("#InputSearch").autocomplete({
		delay: 700,
        minLength: 1,
		source: function( request, response ){    
            $.ajax({
                type:'GET',
                url: url + $("#InputSearch").val(),
				cache:false,
                success: function(data){
                    var date = new Date();
					var notFound = true;
					var shownUrlArr = [];
					var count = 0;

                    searchVal = $("#InputSearch").val();
                    $( "#results" ).html("");
                    $( "#footerDiv" ).html("");
                    
                    
                    if (searchVal.length < 1) {
                        $( "#results" ).html("<h4>0 results found</h4>");
                    }
                    else{
						var section = "";
						if(buttonSwitch == "1"){section = "all";}
						if(buttonSwitch == "2"){section = "Services";}
						if(buttonSwitch == "3"){section = "Forms";}
						if(searchVal.length > 2){
							//send to google analytics
							ga('send', 'pageview', 'index.html?search=' + $("#InputSearch").val() + '&section=' + section);
						}
					
                        // load the featured items first that match the search
						if (buttonSwitch == "1"){
							for (var i = 0; i < featuredArr.length; i++) {
                                if(count == 5){
                                    break;
                                }
                                var str = featuredArr[i].keywords; 
                                var ext = featuredArr[i].external; 
                                var re = new RegExp(searchVal,"ig");
                                var linkLine = "<div class='seSolr_title'><a href='" + featuredArr[i].link  + "'>&#9734; &nbsp;" + featuredArr[i].title + "</a></div>";

                                if (str.match(re)) { 
                                    if (ext == "EXTERNAL")
                                        linkLine = "<div class='seSolr_title'><a href='" + featuredArr[i].link  + "' target='_blank'>&#9734; &nbsp;" + featuredArr[i].title + "</a></div>";

                                    $('#results').append("<div id='div" + count + "'class='divBox'></div>"); 
                                    $('#div' + count).append(linkLine);
                                    $('#div' + count).append("<div class='seSolr_url'>" + featuredArr[i].link + "</div>");
                                    $('#div' + count).append("<div class='seSolr_summary'>" + featuredArr[i].desc + "</div>");

                                    shownUrlArr.push({ "url" : featuredArr[i].link });

                                    count++; 
                                }			
                            }
				        }
                        
                        //load results from se search
                        $(data).filter("div").each(function() {  
                            if (count <= 5) {
								$(this).html($(this).html().replace("&amp;","&"));
								if ($(this).hasClass("seSolr_title")) {
									var text = $(this).html();
									var urls = text.match(/"((?:\\.|[^"\\])*)"/); // get url
									notFound = true;

									//look for duplicate links already shown to bypass
									for (var x = 0; x < shownUrlArr.length; x++) {
										if (urls[1] == shownUrlArr[x].url ) {
											notFound = false;
											break;
										}
                                        if(buttonSwitch == "1"){
                                          var matches = urls[1].match( /\/event|-alert|forms-and-publications|-calendar|.js/g );
										  if (matches != null) {
                                            notFound = false;
											break;
										  }
                                        }
                                        
									}
									
									if (notFound) {
										$('#results').append("<div id='div" + count + "'class='divBox'></div>"); 
										$('#div' + count).append(this);
									}
                                }

								if(notFound) {
									if ($(this).hasClass("seSolr_url")) { 
										$('#div' + count).append(this); 
									}
                                    if ($(this).hasClass("seSolr_summary")) {
										$('#div' + count).append(this); 
										$('#results').append($('#div' + count)); 
										count++; 
									}
                                }

                            }
                        });

                        if (count < 1) {
                            $('#results').append("<h4>0 results found</h4>");
                        }
                        else{
							var feedback = "<a class='icon-email-grey' style='font-weight:normal;' href='http://www.aacounty.org/feedback-form/search-feedback/index.html?keyword=" + $("#InputSearch").val() +	"' >Search Feedback</a>";
                            if (buttonSwitch == "1"){
                                vSection = "all";
								$('#footerDiv').append("<br><div style='font-size:85%;color:gray;'>&#9734; = Recommended Result <br>" + feedback + "</div><center><a id='Submit' class='btn-blue-arrow' href='http://www.aacounty.org/search-apps/index.html?search=" + searchVal + "&section=" + vSection + "'>More Results</a></center>");
                            }
							if (buttonSwitch == "2"){
                                vSection = "Services";
								$('#footerDiv').append("<br><div style='font-size:85%;color:gray;'>" + feedback + "</div><center><a id='Submit' class='btn-blue-arrow' href='http://www.aacounty.org/search-apps/index.html?search=" + searchVal + "&section=" + vSection + "'>More Results</a></center>");
                            }
							if (buttonSwitch == "3"){
                                vSection = "Forms";
								$('#footerDiv').append("<br><div style='font-size:85%;color:gray;'>" + feedback + "</div><center><a id='Submit' class='btn-blue-arrow' href='http://www.aacounty.org/search-apps/index.html?search=" + searchVal + "&section=" + vSection + "'>More Results</a></center>");                             
                            }
                        }
                    }
                    
                    count = 0;

                    $('#results').hide().fadeIn(500,function(){$("#loading").hide();}); 
                },
                error: function(xhr, textStatus, errorThrown){
                  console.log('request failed:' + textStatus);
                }
            });
        }
    });
