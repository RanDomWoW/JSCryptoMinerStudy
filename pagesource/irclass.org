            e.preventDefault();

            var Username = $('#UserName1').val();

                url: url,

                dataType: 'json',

                    var result = data.retToken;

                    $("#SuccessDataError").addClass("hidden");

    function readCookie(name) {
        var nameEQ = name + "=";
        var ca = document.cookie.split(';');
        for (var i = 0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') c = c.substring(1, c.length);
            if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
        }
        return '';
    }

            $('#idEnquiryEmail').focus();
            returnValue = false;
            return returnValue;
        }
        if (!email.match(regexemail)) {
            toastr.error('Please enter valid email', null, { timeOut: 5000 });
            $('#idEnquiryEmail').focus();
            returnValue = false;
            return returnValue;
        }
        var atpos = email.indexOf("@");
        var dotpos = email.lastIndexOf(".");
        if (atpos < 1 || dotpos < atpos + 2 || dotpos + 2 >= email.length) {
            toastr.error('Please enter valid email', null, { timeOut: 5000});
            returnValue = false;
            $('#idEnquiryEmail').focus();
            return returnValue;
        }
        if ($('#idEnquiryMessage').val() == '') {
            toastr.error('Please enter message.', null, { timeOut: 5000});
            $('#idEnquiryMessage').focus();
            returnValue = false;
            return returnValue;
        }
        return returnValue;
    }

    function selectDeselect(listid, status) {
        var listb = document.getElementById(listid);
        var len = listb.options.length;
        for (var i = 0; i < len; i++) {
            listb.options[i].selected = status;
        }
    }
    function validateForm() {
        var returnValue = true;
        var name = $('#idName').val();
        if (name == '') {
            toastr.error('Please enter name', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
            returnValue = false;
            $('#idName').focus();
            return returnValue;
        }
        if (!name.match("^([a-zA-Z]+[,.]?[ ]?|[a-zA-Z]+['-]?)+$"))
        {
            toastr.error('Please enter valid name', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
            returnValue = false;
            $('#idName').focus();
            return returnValue;
        }

        var email = $('#idEmail').val();

        if (email == '') {
            toastr.error('Please enter email', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
            $('#idEmail').focus();
            returnValue = false;
            return returnValue;
        }
        var atpos = email.indexOf("@");
        var dotpos = email.lastIndexOf(".");
        if (atpos < 1 || dotpos < atpos + 2 || dotpos + 2 >= email.length) {
            toastr.error('Please re-enter valid email', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
            $('#idEmail').focus();
            returnValue = false;
            return returnValue;
        }
        if ($('#bSeg :selected').length == 0) {
            toastr.error('Please select business segment.', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
            $('#bSeg').focus();
            returnValue = false;
            return returnValue;
        }
        return returnValue;
    }
    function subscribe() {
        if (validateForm() == true) {
            var selectedString = "";
            var countValue = 0;
            var email = $('#idEmail').val();
            var name = $('#idName').val();
            var selectedValue = '';
            if ($('#bSeg :selected').length > 0) {
                $('#bSeg :selected').each(function (i, selected) {
                   
                    if (countValue == 0) {
                        selectedString = $(selected).text();
                        selectedValue = $(selected).val() + '#' + $(selected).text();
                    }
                    else {
                        selectedString = selectedString + "," + $(selected).text();
                        selectedValue = selectedValue + "," + $(selected).val() + '#' + $(selected).text();
                    }
                   
                    countValue = countValue + 1;
                });

                $("#Service").val(selectedString);
                var modelData = { 'Name': name, 'Email': email, 'Service': selectedString, 'isFail': false, 'MesssagesFailSuccess': '', 'ServiceValue': selectedValue }
                $.ajax({
                    url: '/umbraco/Surface/Home/Subscribe',
                    type: "POST",
                    data: JSON.stringify(modelData),
                    contentType: 'application/json; charset=utf-8',
                    dataType: 'json',
                    success: function (result) {
                        if (Boolean(result.IsFail) == true) {
                            toastr.error(result.MesssagesFailSuccess, null, { timeOut: 5000, positionClass: "toast-bottom-right" });
                            $('#idEmail').val('');
                            $('#idName').val('');
                            selectDeselect('bSeg', false);
                        }
                        else if (Boolean(result.IsFail) == false) {
                            toastr.success(result.MesssagesFailSuccess, null, { timeOut: 5000, positionClass: "toast-bottom-right" });
                            $('#idEmail').val('');
                            $('#idName').val('');
                            selectDeselect('bSeg', false);
                        }
                        else {
                            toastr.error('Please re-enter values.', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
                            $('#idEmail').val('');
                            $('#idName').val('');
                            selectDeselect('bSeg', false);

                        }
                        $('.multi-select').multiselect("refresh");
                    },
                    error: function (result) {
                        console.log(result)
                        if (Boolean(result.IsFail) == true) {
                            toastr.error(result.MesssagesFailSuccess, null, { timeOut: 5000, positionClass: "toast-bottom-right" });
                            $('#idEmail').val('');
                            $('#idName').val('');
                            selectDeselect('bSeg', false);
                        }
                        else {
                            toastr.error('Please re-enter values.', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
                            $('#idEmail').val('');
                            $('#idName').val('');
                            selectDeselect('bSeg', false);
                        }
                        $('.multi-select').multiselect("refresh");

                    }
                });
            }
            else {
                toastr.error('Please select business segment.', null, { timeOut: 5000, positionClass: "toast-bottom-right" });
                $("#bSeg").focus();
            }

        }

    };
