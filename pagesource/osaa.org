                $('input[data-action="search"]').on('keypress', function(event)
                {

                    if (keycode == '13')

                    searchSite();

                function searchSite()
                {                    
                    var term = $('input[data-action="search"]').val();
                    if (term == "")
                    {
                        return false;
                    }

                    var root = "https://www.google.com/search";
                    var query = "?q=site:www.osaa.org ";                    

                    var url = root + query + term;
                    
                    console.log(url);

                    window.open(url);
