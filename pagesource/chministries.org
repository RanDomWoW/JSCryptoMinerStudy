    <script>
        function checkedYou() {
            var x = document.getElementById("chkYou").checked;

            if (x == true) {
                document.getElementById("sltYou").value = "150";
                document.getElementById("sltYou").disabled = false;
            }

            if (x == false) {
                document.getElementById("sltYou").value = "0";
                document.getElementById("sltYou").disabled = true;
            }

            calcFee();
        }

        function checkedSpouse() {
            var x = document.getElementById("chkSpouse").checked;

            if (x == true) {
                document.getElementById("sltSpouse").value = "150";
                document.getElementById("sltSpouse").disabled = false;
            }

            if (x == false) {
                document.getElementById("sltSpouse").value = "0";
                document.getElementById("sltSpouse").disabled = true;
            }

            calcFee();
        }

        function checkedDependent() {
            var x = document.getElementById("chkDependent").checked;

            if (x == true) {
                document.getElementById("sltDependent").value = "150";
                document.getElementById("sltDependent").disabled = false;
            }

            if (x == false) {
                document.getElementById("sltDependent").value = "0";
                document.getElementById("sltDependent").disabled = true;
            }

            calcFee();
        }

        function lvlYou(x) {
            if (x == 0) {
                document.getElementById("chkYou").checked = false;
                document.getElementById("sltYou").disabled = true;
            }

            calcFee();
        }

        function lvlSpouse(x) {
            if (x == 0) {
                document.getElementById("chkSpouse").checked = false;
                document.getElementById("sltSpouse").disabled = true;
            }

            calcFee();
        }

        function lvlDependent(x) {
            if (x == 0) {
                document.getElementById("chkDependent").checked = false;
                document.getElementById("sltDependent").disabled = true;
            }

            calcFee();
        }

        function calcFee() {
            var x;

            x = parseInt(document.getElementById("sltYou").value) + parseInt(document.getElementById("sltSpouse").value) + parseInt(document.getElementById("sltDependent").value);

            document.getElementById("fee").innerHTML = "Monthly Program Fee:  $" + x;

            document.getElementById("chkBK").disabled = false;

            if (x == 0) {
                document.getElementById("chkBK").checked = false;
                document.getElementById("chkBK").disabled = true;

                document.getElementById("bkAnnual").innerHTML = "";

                document.getElementById("bkQuarterly").innerHTML = "";
            }

            checkedBK();
        }

        function checkedBK() {
            var x = document.getElementById("chkBK").checked;
            var count = 0;

            if (x == true) {
                document.getElementById("bkAnnual").innerHTML = "Annual Brother's Keeper Fee:  $40";

                if (document.getElementById("chkYou").checked == true) {
                    count = count + 1;
                }

                if (document.getElementById("chkSpouse").checked == true) {
                    count = count + 1;
                }

                if (document.getElementById("chkDependent").checked == true) {
                    count = count + 1;
                }

                count = count * 25;

                document.getElementById("bkQuarterly").innerHTML = "Quarterly Estimated Fee:  $" + count;
            }

            if (x == false) {
                document.getElementById("bkAnnual").innerHTML = "";

                document.getElementById("bkQuarterly").innerHTML = "";
            }
        }

        window.onload = function () {
            calcFee();
        }
    </script>
