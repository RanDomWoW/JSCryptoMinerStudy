function search(frm){
  var hashtag = frm.search_box.value
  //remove duplicate whitespace
  hashtag = hashtag.replace(/\s+/g,' ');
  //trim spaces front and back
  hashtag = hashtag.replace(/^\s+|\s+$/g,'')
  //remove all non alphanumeric charts, but leave comma, underscore, and space
  hashtag = hashtag.replace(/[^a-z0-9\,_\s\u0080-\uffff]/gi,'');
  //replace remaining spaces with a comma
  hashtag = hashtag.replace(/\s/g,',');
  //replace multi commas with a comma
  hashtag = hashtag.replace(/,+/g,',');

  if (typeof hashtag !== 'undefined' && hashtag.length>0)
    window.location =  '/analytics/' + hashtag + '/';
  else
  {
    frm.search_box.value = ""
    alert('Please enter a valid search term');
  }
  return false; 
}

      Simply put, a hashtag is an easy way for people to categorize, find and join conversations on a particular topic.

The hash mark, or pound symbol, (#) is now known by social media users as a "hashtag" or "hash tag". Posts that have the same keywords prefixed with the # symbol are grouped together in user searches, bringing a level of order to the frenzied chaotic world of Twitter.
