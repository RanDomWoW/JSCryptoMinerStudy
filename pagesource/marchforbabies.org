    document.addEventListener('DOMContentLoaded', function() {

        const bellIcon = document.querySelector('.bell-icon');
        const nc = document.querySelector('.notifications-control');
        const menuDropDown = document.querySelectorAll('.menu-content li.has-sub > a');
        let lastScrollTop = 0;
        let userControlOpen = false;
        let notificationsOpen = false;

        // Desktop Specific
        if (window.innerWidth >= 800) {
            // Masthead hide / reveal
            document.addEventListener('scroll', function() {
                const st = window.pageYOffset || document.documentElement.scrollTop;

                if (st > 120) {
                    if (st < lastScrollTop) {
                        // Sticky
                        masthead.style.transform = 'translateY(-70px)';
                        logo.style.transform = 'translateY(-55px) scale(.5)';
                        logo.style.position = 'relative';
                        logo.style.top = '90px';

                        if (userNav) {
                            userNav.style.transform = 'translateY(-70px)';
                        }
                    } else {
                        // Hidden
                        masthead.style.transform = 'translateY(-150px)';
                        logo.style.position = 'static';

                        if (userNav) {
                            userNav.style.transform = 'translateY(-225px)';
                        }
                    }

                    lastScrollTop = st;
                } else {
                    // Normal
                    masthead.style.transform = 'translateY(0px)';
                    logo.style.position = 'static';
                    logo.style.transform = 'translateY(0px) scale(1)';

                    if (userNav) {
                        userNav.style.transform = 'translateY(0px)';
                    }
                }
            });

            // Util Nav dropdown listeners
            document.addEventListener('click', function(e) {
                // Check if 'user' element is present
                if (user) {
                    if (e.target === user || e.target === userArrow) {
                        if (!userControlOpen) {
                            document.querySelector('.user-control').style.display = 'flex';
                            userControlOpen = !userControlOpen;
                        } else {
                            document.querySelector('.user-control').style.display = 'none';
                            userControlOpen = !userControlOpen;
                        }
                    } else {
                        document.querySelector('.user-control').style.display = 'none';
                    }

                    if (e.target === notifications || e.target === counter || e.target === bellIcon || e.target === nc) {
                        if (!notificationsOpen) {
                            const coords = notifications.getBoundingClientRect();

                            nc.style.display = 'flex';
                            nc.style.left = coords.left - 240 + 'px';
                            nc.style.top = '52px';
                            notificationsOpen = !notificationsOpen;
                        } else {
                            nc.style.display = 'none';
                            notificationsOpen = !notificationsOpen;
                        }
                    } else {
                        nc.style.display = 'none';
                    }
                }
            });
        }

        // Mobile Specific
        if (window.innerWidth < 800) {
            // Masthead hide / reveal
            document.addEventListener('scroll', function() {
                const st = window.pageYOffset || document.documentElement.scrollTop;

                if (st > 95) {
                    if (st < lastScrollTop) {
                        masthead.style.position = 'fixed';
                        masthead.style.transform = 'translateY(0px)';
                    } else {
                        masthead.style.transform = 'translateY(-100px)';
                    }

                    lastScrollTop = st;
                }
            });

            // Mobile Navigation - Sub Navigation Dropdown
            menuDropDown.forEach(function(e) {
                e.addEventListener('click', function(e) {
                    const el = e.target.parentNode;

                    if (el.classList.contains('open')) {
                        el.classList.remove('open');
                        e.target.nextElementSibling.style.display = 'none';
                    } else {
                        el.classList.add('open');
                        e.target.nextElementSibling.style.display = 'block';
                    }
                });
            });

            // Mobile Navigation - Hamburger Menu
            openNav.addEventListener('click', function() {
                mobileNav.style.transform = 'translateX(0)';
            });

            closeNav.addEventListener('click', function() {
                mobileNav.style.transform = 'translateX(100vw)';
            });

                mobileNots.style.transform = 'translateY(0)';
            });

            closeNots.addEventListener('click', function() {
                mobileNots.style.transform = 'translateY(-100vh)';
            });
        }
    });

                currentSlide += 1;

        var zipGeo = "";

        function nearestEventFail() {
            document.querySelector('.no-location').style.display = 'flex';
            document.querySelector('.location').style.display = 'none';
        }

        function findNearestEvent(zip) {
            if (zip == null || typeof(zip) == "undefined" || zip == "") {
                nearestEventFail();
                return;
            }

            $.ajax({ url: "/Registration/Events_GET_Ajax", method: "POST", data: { term: zip, distance: 100, start: 0, count: 1 } })
                .done(function(data) {
                    // Total, List (EventInfo), Location (string)
                    if (data.Total > 0) {
                        var event = data.List[0];
                        var eventDateLocation = event.EventDateAsString + ", " + event.City + ", " + event.State;

                        // let's change le DOM!
                        document.querySelector('.no-location').style.display = 'none';
                        document.querySelector('.location').style.display = 'flex';
                        document.querySelector('.march-name').textContent = event.SiteName;
                        document.querySelector('.march-date-location').textContent = eventDateLocation;
                        document.querySelector('.march-button').setAttribute('href', '/Registration/Signup?eventId=' + event.EventID);
                    }
                    else {
                        nearestEventFail();
                    }
                })
            .fail(function (jqXHR, textStatus, errorThrown) {
                nearestEventFail();
            })
            .always(function () {
            });
        }

        document.addEventListener('DOMContentLoaded', function() {
            BrowserGeolocationHelper.Init(
                function() {
                    // success - use geo-zip
                    zipGeo = BrowserGeolocationHelper.GetZip();
                    if (zipGeo != null && zipGeo !== undefined && zipGeo != "") findNearestEvent(zipGeo);
                    else findNearestEvent(zipByIp);
                },
                function() {
                    nearestEventFail();
                }
            );
        });
