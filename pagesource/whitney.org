        $(document).ready(function () {
            window.isMobile = !!(window.innerWidth < 1000);
            var w = 'center'
            var text = document.getElementById('w__text');
            var line = document.getElementById('w__line');
            var svg = document.getElementById('w__new');
            var svgMobile = document.getElementById('w__new-mobile');
            var primaryNav = document.getElementsByClassName('primary-nav')[0];
            var primaryNavList = document.getElementsByClassName('primary-nav__list')[0];
            var miniTitle = document.getElementsByClassName('w-mini-header__title')[0];

            var marginHeight = 20;
            var activeMarginHeight = marginHeight;
            var maxDesktopHeight = 160;
            var maxMobileHeight = 140;
            var maxHeight = maxDesktopHeight;
            var basePadding = 10;
            var adjustmentFactor = .96;

            var navWidth;
            var maxWidth;
            var newLeft = 'auto';
            var newRight = 'auto';
            var offsetY = 0;
            var ratio = 6; // 119 : 20 ~= 6 : 1
            var ticking = false;

            function update() {
                ticking = false;
                var newHeight = Math.max(0, maxHeight - offsetY);

                if (offsetY >= maxHeight + activeMarginHeight) {
                    primaryNav.classList.add('primary-nav--fixed');
                    miniTitle.classList.add('w-mini-header__title--active');
                    line.classList.add('inactive');

                } else {
                    primaryNav.classList.remove('primary-nav--fixed');
                    miniTitle.classList.remove('w-mini-header__title--active');
                    line.classList.remove('inactive');
                }

                newHeight == 0 ? line.style.display = 'block' : line.style.display = 'none';
                window.isMobile ? svgMobile.style.height = newHeight + 'px' : svg.style.height = newHeight + 'px';

                drawText();
            };

            function drawText() {
                var height = maxHeight - offsetY;
                var newHeight = 0;
                var newWidth = 0;

                if (height > 0) {
                    var width = maxWidth / 2;
                    newHeight = (height * width) / (width + ratio * height) * .99;
                    newWidth = ratio * newHeight * .99;

                    if (w == 'left') {
                        newWidth = newWidth * adjustmentFactor;
                        newLeft = basePadding * (height / maxHeight) + 2 + 'px';

                    } else if (w == 'right') {
                        newWidth = newWidth * adjustmentFactor;
                        newRight = (maxWidth / 2) + basePadding * (height / maxHeight) + 2 + 'px';

                    } else {
                        newLeft = (maxWidth / 4) - (newWidth / 2) + 2 + 'px';
                    }
                }

                Object.assign(text.style, {
                    left: newLeft,
                    right: newRight,
                    width: newWidth + 'px',
                    height: newHeight + 'px'
                });
            }

            function setup() {
                // Set size specific values and adjustments
                if (window.isMobile) {
                    navWidth = window.innerWidth - 20;
                    maxHeight = maxMobileHeight;
                    activeMarginHeight = 0;
                    basePadding = navWidth / 80;
                    adjustmentFactor = .97;

                } else {
                    navWidth = primaryNavList.offsetWidth;
                    maxHeight = maxDesktopHeight;
                    activeMarginHeight = marginHeight;
                    basePadding = navWidth / 100;
                    adjustmentFactor = .96;
                }

                offsetY = window.scrollY;
                maxWidth = navWidth;

                update();
                drawText();

                window.isMobile ? svgMobile.style.display = 'block' : svg.style.display = 'block';
                Object.assign(text.style, {right: newRight, display: 'block'});
            }

            function onScroll() {
                offsetY = window.scrollY;
                requestTick();
            }

            function requestTick() {
                if (!ticking) {
                    requestAnimationFrame(update);
                }

                ticking = true;
            }

            $(window).resize(function () {
                setup();
            });

            window.addEventListener('scroll', onScroll, false);
            setup();
        });
    </script>
