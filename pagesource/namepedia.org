function moveIn() {
	if (movedY < (diffY / 2)) {
		stepY++;
	} else {
		stepY--;
		if (stepY < 0) {
			endMoveIn();
			return;
		}
	}
	movedY += stepY;
	var donePercent = movedY * 100 / diffY;
	searchBoxElm.style.left = startLeft - (diffX / 100 * donePercent);
	searchBoxElm.style.top  = startTop + movedY;
	if (movedY >= diffY) {
		endMoveIn();
	}
}
function startMoveIn() {
	//searchBoxElm.style.border = '1px solid gray';
	document.getElementById('textSearch').focus();
	moveInterval = window.setInterval(moveIn, 15);
}
function endMoveIn() {
	window.clearInterval(moveInterval);

}
