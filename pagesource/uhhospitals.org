    $(document).ready(function () {
	var typingTimer;                //timer identifier
	var doneTypingInterval = 1000;  //time in ms (5 seconds)

	//on keyup, start the countdown
	$('.search-box-text').keyup(function(){
	    clearTimeout(typingTimer);

        $(document).on('click', '.suggestion-link', function (e) {
            e.preventDefault();
            $(".search-box-text").val($(this).text());
            $(".autocomplete-dd").hide();
            window.location = $(this).attr("href");
            return false;
        });

        $(document).on('click', '.search-button', function (e) {
            e.preventDefault();
            window.location = '/search?q=' + $('.search-box-text').val();
        });

        function AutosuggestPost() {
		var searchValue = $('.search-box-text').val();
            $.ajax({
                url: "/bingautocomplete.aspx",
                method: "POST",
                data: { query: searchValue },
                success: function (response) {
                    SetSuggestions(response);
                }
            });
        }

        function SetSuggestions(suggestions) {
            var bingSuggestions = suggestions.split('|');
            var output = '';
            if (bingSuggestions != 'No Results') {
                $.each(bingSuggestions, function (key, suggestion) {
                    output += '<li><a href="/search?q=' + suggestion + '" class="suggestion-link">' + suggestion + '</a></li>';
                });
                if (output != '') {
                    output = "<ul>" + output + "</ul>";
                    $(output).appendTo('.autocomplete-dd');
                }
            }
        }
    });
