//returns an array with all values for the key 'name' in a URI

function getParametersByName(name) {

        var stringToBeMatched = window.location.search;

        var patternToBeMatched = new RegExp("[?&]" + name + "([^&]*)", "g");

        var patternToBeReplaced = new RegExp("[?&]" + name + "=", "g");

        var match = stringToBeMatched.match(patternToBeMatched);

        if (match) {

            //go through array and remove the tab identifier

            for (var i = 0; i < match.length; i++) {

                match[i] = match[i].replace(patternToBeReplaced, '');

            }

        }

        return match;

    } //end getParametersByName(name)



function checkQueryString(queryStringField) {

        //an array is returned from getParametersByName

        var tabsQueryString = getParametersByName(queryStringField);

        if (tabsQueryString) {

            //for each instance in tabsQueryString array, change the contentDocument

            for (var i = 0; i < tabsQueryString.length; i++) {

                switchContentAndTabs(tabsQueryString[i]);

            }

        }

    } //end checkQueryString (queryStringField)



function switchContentAndTabs(tabsContentId) {

        //identify the parent id for the tabs

        //tabsContentId may contain two strings (two classes, the second one may be tabsActive)

        var mySplitContentId = tabsContentId.split(" ");

        var tabsContentId = mySplitContentId[0];

        var tabsParentId = jQuery('#' + tabsContentId).parent().attr('id');

        //alert("tabsContentId: "+tabsContentId+" tabsParentId: "+tabsParentId);

        // if the tabsContentId does not exist on the page OR is currently active, return immediately.

        if (tabsParentId && !(jQuery('#' + tabsParentId + ' .tabsNavigation a.' +

            tabsContentId).hasClass('tabsActive')) && (!(jQuery('.' +

            tabsContentId).hasClass('tabsActive')) || jQuery('#' +

            tabsParentId).attr('role') != 'openAccordion')) {

            // alert('firstif');

            // !(jQuery('#'+tabsParentId+ ' .tabsContentDiv').attr('style') == 'display: block;');

            //update the list's active class

            jQuery('#' + tabsParentId + ' .tabsNavigation a').removeClass(

                'tabsActive');

            jQuery('#' + tabsParentId + ' .tabsNavigation a.' + tabsContentId).addClass(

                'tabsActive');

            //update the visible content div

            if ((jQuery('#' + tabsParentId).hasClass('course-details'))) {

                //(jQuery('#'+tabsParentId).attr('role') == 'openAccordion') && 

                jQuery('.' + tabsContentId).addClass('tabsActive');

                //alert('course_detail');

            }

            //determine the animation requested: show (default), fadeIn, slideDown

            if (jQuery('#' + tabsParentId).hasClass('tabsAnimationFadeIn')) {

                //For ANIMATION = FADE IN

                jQuery('#' + tabsParentId + ' .tabsContentDiv').hide();

                jQuery('#' + tabsParentId + ' #' + tabsContentId).fadeIn("slow");

            } else if (jQuery('#' + tabsParentId).hasClass(

                'tabsAnimationSlideDown')) {

                //For ANIMATION = SLIDE UP AND DOWN

                //jQuery('div#'+tabsParentId+' div').hide();

                jQuery('#' + tabsParentId + ' .tabsContentDiv').slideUp();

                jQuery('#' + tabsParentId + ' #' + tabsContentId).slideDown(

                    "slow");

            } else {

                //For ANIMATION = IMMEDIATE

                jQuery('#' + tabsParentId + ' .tabsContentDiv').hide();

                jQuery('#' + tabsParentId + ' #' + tabsContentId).show();

            }

        }

        // if the tabsContentId does exist on the page  and is currently active and Role is OpenAccordion, assume double click.

        else if (tabsParentId && (

            (

                (jQuery('#' + tabsParentId + ' .tabsNavigation a.' +

                    tabsContentId).hasClass('tabsActive')) && (jQuery('#' +

                    tabsParentId).attr('role') == 'openAccordion')) || (

                (jQuery('.' + tabsContentId).hasClass('tabsActive')) && (

                    jQuery('#' + tabsParentId).hasClass('course-details')))

        ))

        //(jQuery('#'+tabsParentId).hasClass('course-details'))

        {

            //hide the currently open content

            //alert('Hide Accordion'); 

            //alert(jQuery('div#'+tabsParentId+' .tabsNavigation a.'+tabsContentId).hasClass('tabsActive'));

            //alert('tabsParentId '+tabsParentId+' tabsContentId '+tabsContentId);

            var varClass = jQuery('#' + tabsParentId + ' .tabsNavigation a.' +

                tabsContentId).attr('class');

            //alert(varClass);

            //update the list's active class

            //jQuery('#'+tabsParentId+' .tabsNavigation a').removeClass('tabsActive');

            if ((jQuery('#' + tabsParentId).attr('role') == 'openAccordion')) {

                jQuery('#' + tabsParentId + ' a').removeClass('tabsActive');

            }

            //jQuery('#'+tabsParentId+' .tabsNavigation a.'+tabsContentId).addClass('tabsActive');

            //update the visible content div

            //determine the animation requested: show (default), fadeIn, slideDown

            if (jQuery('#' + tabsParentId).hasClass('tabsAnimationFadeIn')) {

                //For ANIMATION = FADE IN

                jQuery('#' + tabsParentId + ' .tabsContentDiv').fadeOut("slow");

                //jQuery('#'+tabsParentId+' #'+tabsContentId).fadeIn("slow");

            } else if (jQuery('#' + tabsParentId).hasClass(

                'tabsAnimationSlideDown')) {

                //For ANIMATION = SLIDE UP AND DOWN

                jQuery('#' + tabsParentId + ' .tabsContentDiv').slideUp();

                //jQuery('div#'+tabsParentId+' div.tabsContentDiv').slideUp();

                //jQuery('#'+tabsParentId+' #'+tabsContentId).slideDown("slow");	

            } else {

                //For ANIMATION = IMMEDIATE

                jQuery('#' + tabsParentId + ' .tabsContentDiv').hide();

                //jQuery('#'+tabsParentId+' #'+tabsContentId).show();

            }

        } else {

            //do nothing

        }

    } //end switchContentAndTabs (tabsContentId)
