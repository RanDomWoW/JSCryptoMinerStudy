                                if(2 < promo_cards)
                                {
                                    $(first_card).stop(true,true);
                                    clearTimeout(slider);
                                }
                                setCurrentPromotion(prev_card);
                                setNavDot(prev_card);
                                glide(last_card,'left',1000);
                                if(2 < promo_cards)

                                    slider = setAutoGlide();
                                }

                                var promo_card = $('.promotional_card_container[data-promo_number=promo_'+dot_num+']');

                                {
                                    $(first_card).stop(true,true);
                                    clearTimeout(slider);
                                }
                                $('.promo_dot_nav').removeClass('nav_select');
                                $(this).addClass('nav_select');

                                    rebuildImages(promo_card, dot_num, 1);
                                }
                                else if(3 < promo_cards)
                                {
                                    rebuildImages(promo_card, dot_num, 2);
                                }
                                else
                                {
                                    $(promo_card).prependTo('.hp_promo_main_container');
                                }

                                {
                                    slider = setAutoGlide();
                                }

                                $('.promotional_card_container').removeClass('current_promotion');

                            {

                                $('.promo_dot_nav').removeClass('nav_select');
                                $('.promo_dot_nav[data-promo_nav=dot_'+id+']').addClass('nav_select');
                            }
                            
                            function isDoubleClicked(element) {
                                //if already clicked return TRUE to indicate this click is not allowed
                                if (element.data("isclicked")) return true;

                                //mark as clicked for 1 second
                                element.data("isclicked", true);
                                setTimeout(function () {
                                    element.removeData("isclicked");
                                }, animation_timing);

                                //return FALSE to indicate this click was allowed
                                return false;
                            }

                            {

                                switch(direction)

                                    case 'right':

                                            $(item).animate({width: "20.0000%"}, 0);

                                        $(item).animate({width: "20.0000%"}, duration, function(){

                                        });

                                }

                                setNavDot(next_card);
                                glide(first_card,'right',2000);
                                slider = setAutoGlide();
                            }

                                }

                            }

			function is_ie() 
			{
				var ua = window.navigator.userAgent;
				var msie = ua.indexOf("MSIE ");

				if (msie > 0 || !!navigator.userAgent.match(/Trident.*rv\:11\./))  // If Internet Explorer, return version number
				{
					return true;
				}
				else  // If another browser, return 0
				{
					return false;
				}

				return false;
			}
			
            function log_form_click(form_name, form_link, action_anchor)
			{
                var data = {
                    request_type:'log_form_request',
                    form_name:form_name,
                    form_link:form_link
                };
                            
				$.ajax({
					type: 'POST',
					url: '?ajax=1',
					data: data,
                    complete: function(){
                        var action_url = action_anchor.attr('href');
                        var action_location = action_anchor.attr('target'); 
                        if('_blank' === action_location)
                        {
                            window.open(action_url);
                            return true;
                        }
                        else 
                        {
                            window.location = action_url;
                            return false;
                        }
                    }
				});
			}

			{
