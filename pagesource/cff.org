            /**
             * loop through instagram links and replace with embedded instagram post
             * will find all references of <a class=instagram-embed" href="instagram.post.here"></a>
             *
             * Dependencies: 
             *     jQuery 
             *     instgrm embed script https://platform.instagram.com/en_US/embeds.js
             *
             * Author: Christopher Lindsey
             */

            (function embedInstagram(window, $, instgrm, undefined) {
                var numEmbeds = $('a.instagram-embed').length;
                var numEmbedsLoaded = 0;

                $(function () {
                    $('a.instagram-embed').each(function (index, element) {
                        embed(element);
                    });
                });

                function embed(element) {
                    var $el = $(element),
                        url = $el.attr('href');

                    $.ajax({
                        type: "GET",
                        url: "https://api.instagram.com/oembed/?url=" + url + "&omitscript=true",
                        dataType: "jsonp"
                    })
                    .done(function (data) {
                        $el.after(data.html).css('display', 'none');
                        numEmbedsLoaded++;
                        if (numEmbedsLoaded >= numEmbeds
                            && typeof (instgrm) !== typeof (undefined)) {
                            instgrm.Embeds.process();
                        }
                    });
                }
            })(window, window.jQuery, window.instgrm);
