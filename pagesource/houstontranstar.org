        function getWeatherData()
        {
            $.getJSON("http://api.openweathermap.org/data/2.5/weather?q=Houston", function (result) {
                
                var weatherInfo = {
                    temp: TranStarUtil.calcDegreeFromKelvin(result.main.temp),
                    humidity:  result.main.humidity,
                    windSpeed: Math.round(result.wind.speed),
                    degree: "F",
                    windDir: TranStarUtil.degToCompass(result.wind.deg),
                    windSpeedType: "mph",
                    description: result.weather[0].main,
                    description2: result.weather[0].description,
                    location: result.name,
                    sunset: TranStarUtil.calcEpochTime(result.sys.sunset),
                    weatherIconUrl: "http://openweathermap.org/img/w/" + result.weather[0].icon + ".png"
                };

                var template =
                    "<i class='pull-right fa fa-external-link-square'></i><p>Currently for " + weatherInfo.location + ", TX</p>" + 
                    "<h3><img src='" + weatherInfo.weatherIconUrl + "'/>" + weatherInfo.temp + "&deg; " + weatherInfo.degree + "</h3>" + 
                    "<p><strong>" + TranStarUtil.capFirstLetter(weatherInfo.description2) + "</strong></p><hr/>" + 
                    "<p>Wind " + weatherInfo.windDir + " @ " + weatherInfo.windSpeed + " " + weatherInfo.windSpeedType + "<br />Sunset " + weatherInfo.sunset + "</p>";
                $("#currentWeather").html(template);
            });
        }

        function getNoaaWeatherData() {


            $.ajax({

                url: "http://forecast.weather.gov/MapClick.php?lat=29.77&lon=-95.37&FcstType=json",
                jsonp: "callback",
                dataType: "jsonp",

                success: function (result) {

                    console.log(result);
                    var weatherInfo = {
                        temp: result.currentobservation.Temp,
                        humidity: result.currentobservation.Relh,
                        windSpeed: result.currentobservation.Winds,
                        degree: "F",
                        windDir: TranStarUtil.degToCompass(result.currentobservation.Windd),
                        windSpeedType: "mph",
                        description: result.currentobservation.Weather,
                        description2: result.data.text[0].split('. ')[0],
                        location: result.productionCenter,
                        forecastPeriod: result.time.startPeriodName[0],
                        weatherIconUrl: result.data.iconLink[0]
                    };
                    var template =
                    "<p><strong>" + weatherInfo.location + "</strong></p>" +
                    "<h3><img height='45' width='45' style='margin: 5px;' src='" + weatherInfo.weatherIconUrl + "'/>" + weatherInfo.temp + "&deg; " + weatherInfo.degree + "</h3>" +
                    "<p>" + weatherInfo.description + "<br/>Humidity: " + weatherInfo.humidity + "%<br/>Wind: " + weatherInfo.windDir + " @ " + weatherInfo.windSpeed + " " + weatherInfo.windSpeedType + "<hr/>" +
                    "<p><strong>" + weatherInfo.forecastPeriod + "'s Travel Forecast</strong></p><p>" +  weatherInfo.description2 + "</p>";
                    $("#currentWeather").html(template);
                },
                error: function () {
                    $("#currentWeather").html("Could not load weather conditions.");
                }

            });

        }
