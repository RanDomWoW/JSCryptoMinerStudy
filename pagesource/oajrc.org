     function onLogin() {
         var u = $("#username").val();
         var p = $("#password_text").val();
         var hidtype = $("#hidtype").val();
         if (u == "" || u == "请输入用户名") {
             alert("请输入用户名!");
             return;
         }
         if (p == "" || p == "请输入密码") {
             alert("请输入密码!");
             return;
         }

         if (u != "" && p != "") {
             $.post("/ajax/login", { 'username': u, 'password': p, 'type': hidtype }, function (data) {
                 if (data.success) {
                     if ($("#chk1").attr("checked")) {
                         setCookie("AutoLoginUserName", u, 168, "/");
                         setCookie("AutoLoginPwd", p, 168, "/");
                     }
                     else {
                         deleteCookie("AutoLoginUserName", "/");
                         deleteCookie("AutoLoginPwd", "/")
                     }
                     window.location.href = data.url;
                 } else {
                     alert(data.msg);
                     //$.m.alert(data.msg);
                 }
             }, "json")
         }
     }
