var map;
var flagpath = "viaf/images/flags/";
var pppath = "viaf/partnerpages/";

function initializeMap() {
  var mapOptions = {
     scrollwheel: false,
    zoom: 2,
    center: new google.maps.LatLng(18,0),
    mapTypeId: google.maps.MapTypeId.ROADMAP
  };

  map = new google.maps.Map(document.getElementById('homemap'),
      mapOptions);

    $.each( viafpartners, function(i, partner) {
      if (partner.lat){
        var myLatlng =  new google.maps.LatLng(partner.lat, partner.long);
        new google.maps.Marker({position: myLatlng, map: map, title: partner.name});
        }
      if (partner.children) {
      	$.each(partner.children, function(i, child){
      		if (child.lat){
      		  var myLatlng =  new google.maps.LatLng(child.lat, child.long);
      		  new google.maps.Marker({position:myLatlng, map:map, title:child.name})
      		}
      	})
      }
     });

     $.each( viafpartners, function(i , partner) {
            var li = document.createElement("li");
            var flagimg = document.createElement("img");
            $(flagimg).attr("src", flagpath + partner.flag);
            $(li).append(flagimg);
            
            var lihref = document.createElement("a");
            $(lihref).attr("href", pppath + partner.link);
            var litext = document.createTextNode(partner.name);
            lihref.appendChild(litext);
            li.appendChild(lihref);
            
            if (partner.children) {
                var childul = document.createElement("ul");
                $(childul).attr("class", "partnerchildlist");
                li.appendChild(childul);
                $.each(partner.children, function (i, child) {
                    var childli = document.createElement("li");
                    var childlitext = document.createTextNode(child.name);
                    childli.appendChild(childlitext);
                    childul.appendChild(childli);
                });
            }
            
            if (partner.test == "n") {
                $("#partnerslist").append(li);
            }
            else { 
                $("#partnerslisttest").append(li);
                document.getElementById("hometestpartlibslist").style.display = "block";
            }
     });
    
    $('#partnerslistdiv').columnize({ columns: 2 });
}
 
google.maps.event.addDomListener(window, 'load', initializeMap);
