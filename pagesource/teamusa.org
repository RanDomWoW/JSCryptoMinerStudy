    /****************************************************************************
      singup()
      Called when submit button is pressed on any signup form

    function signup(name, email, popId) {

        var secondaryLink = 'http://action.teamusa.org/page/s/team';

        //If no secondary link defined, call the signup API
        if (secondaryLink === '') {

            var postData = { Name: name, Email: email }
            var cmsBaseUri = 'https://cms.teamusa.org';

            jQuery.ajax({
                url: cmsBaseUri + '/api/teamusa/Login/TeamUsaSignUp',
                type: 'POST',
                data: postData
            })
            .done(function () {
                jQuery("#popup-window-confirmation-container-" + popId + " .success").show();
                jQuery("#popup-background-confirmation-" + popId).show();
                jQuery("#popup-window-confirmation-container-" + popId).show();
            })
            .fail(function (errorDetails) {
                var div = jQuery("#popup-window-confirmation-container-" + popId + " .failure .more-info");
                var moreInfo = '';

                try {
                    var errorDetailsJson = JSON.parse(errorDetails.responseJSON.Message);

                    for (var i = 0, len = errorDetailsJson.errorDetails.length; i < len; i++) {
                        moreInfo = moreInfo + '</br>' + errorDetailsJson.errorDetails[i].msg;
                    }
                }
                catch (e) {
                    try {
                        moreInfo = "An unexpected error occurred. Error: " + errorDetails.responseJSON.Message + " Error Details: " + errorDetails.responseJSON.ExceptionMessage;
                    }
                    catch (e1) {
                        moreInfo = "An unexpected error occurred. Please try again. Response: " + errorDetails;
                    }
                }

                div.html(moreInfo);

                //Display the popup window elements associated with the particular signup form
                jQuery("#popup-window-confirmation-container-" + popId + " .failure").show();
                jQuery("#popup-background-confirmation-" + popId).show();
                jQuery("#popup-window-confirmation-container-" + popId).show();
            });

        }
        else {
            //Redirect to secondary link
            var linkTarget = '_blank';
            if (linkTarget == '_self') {
                window.location.replace("http://action.teamusa.org/page/s/team?firstname=" + name + "&email=" + email);
            }
            else {
                window.open("http://action.teamusa.org/page/s/team?firstname=" + name + "&email=" + email, linkTarget);
            }
        }
    }

    /****************************************************************************
      closePopupConfirmation()
      Close the confirmation popup window
     ****************************************************************************/
    function closePopupConfirmation() {
        jQuery(".popup-confirmation").hide();
        jQuery(".popup-confirmation .success").hide();
        jQuery(".popup-confirmation .failure").hide();
    }

            function getImgLayoutInfo (mosaicItemData) {

                var cropDirection;
