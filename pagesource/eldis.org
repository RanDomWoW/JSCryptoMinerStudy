          <em><div class="field field-name-field-author field-type-entityreference field-label-hidden"><div class="field-items"><div class="field-item even">Alice Webb</div><div class="field-item odd">Grace Lyn Higdon</div></div></div></em>  </li>

  <div class="views-field views-field-title">        <h3 class="field-content"><a href="https://www.eldis.org/blogpost/bridging-research-policy-gap-africa%E2%80%99s-youth-employment-challenge?position=2&amp;list=86vzfTGCRPMviF5R5sh8K6DLOIn7sH0uM5qYkLHs-Jk">Bridging the research-policy gap: Africa’s youth employment challenge</a></h3>  </div>  
  <div class="views-field views-field-created">        <span class="field-content">16 June 2017</span>  </div>
