function commify(num) {
    var newNum = "";
    var ret = "";
    var count = 0;

    for (var k = num.length-1; k >= 0; k--){
      var oneChar = num.charAt(k);
      if (count == 3) {
        newNum += ",";
        newNum += oneChar;
        count = 1;
        continue;
      }
      else {
        newNum += oneChar;
        count ++;
      }
   }  //but now the string is reversed!

  //re-reverse the string
  for (var k = newNum.length-1; k >= 0; k--){
      var oneChar = newNum.charAt(k);
      ret += oneChar;
  }
  return ret;
}
