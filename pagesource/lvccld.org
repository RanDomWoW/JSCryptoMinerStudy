  var changeAlertVariable = function(n)  {
		$.ajax({
			type: 'post',
			dataType: "json",
			url: "/common.cfc?method=showAlertVariable",
			data: { bShow: parseInt(n) },
			success: function (response) {  
				if (response.show_alert == "1") {
					$(".main-alert").css("display","block");
					$(".alert_show").css("display","none");
				} else {
					$(".main-alert").css("display","none");
					$(".alert_show").css("display","block");
				}
				navigationPosition();
			},
			error: function (response) { //error code here
				alert("Error of alert message box displaying"); 
			}
	   });	  
   }
