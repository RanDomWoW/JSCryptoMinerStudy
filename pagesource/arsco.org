    $(document).ready(function () {

        $('#SearchTxt-Web,#SearchTxt-Mobile').click(function (e) {
            var ids = $(this).attr('id');
            ids = ids.split('-');
            var res1 = $('#Searchbox-' + ids[1]).val().toLowerCase().trim();
            if (res1 != "") {
                if (res1 == '') {
                    $('#Searchbox-' + ids[1]).val(" ");
                    return false;
                }
                else {
                    var url = "/Search/Searchresult";
                    $.get(url, { Searchtext: res1 }, function (data) {

                        if (data == "Success") {
                            window.location.href = "/search";
                        }
                        else {
                            $('#Searchbox-' + ids[1]).val(" ");
                            return false;
                        }
                    });
                }
            }
            else {
                $('#Searchbox-' + ids[1]).val(" ");
                $('#Searchbox-' + ids[1]).focus();
                return false;
            }
        });


        $('#Searchbox-web,#Searchbox-mobile').keypress(function (e) {

            var id = $(this).attr('id');

            if (e.which == 13) {
                var res1 = "";
                if ($('#' + id).val() != "") {
                    res1 = $('#' + id).val().toLowerCase().trim();
                    if (res1 == '') {
                        $('#' + id).val('');
                        return false;
                    }
                    else {
                        var url = "/Search/Searchresult";
                        $.get(url, { Searchtext: res1 }, function (data) {

                            if (data == "Success") {
                                window.location.href = "/search";
                            }
                            else {
                                $('#Searchbox-' + ids[1]).val(" ");
                                return false;
                            }
                        });
                    }
                }
                else {
                    $('#' + id).val('');
                    return false;
                }
            }
        });
    });

    function LoadCategory() {
        $.ajax({
            type: "POST",
            url: "/Home/LoadCategory",
            dataType: "json",
            error: function (xhr, status, error) {
            },
            success: function (data) {
                var HtmlString = "";
                var TotalCount = data.length;
                $.each(data, function (i, e) {
                    if (i == 0) {
                        HtmlString += '<li class="first-slide">';
                    }
                    else if (i == 3) {
                        HtmlString += '<li class="last-slide">';
                    }
                    else {
                        HtmlString += '<li>';
                    }
                    HtmlString += '<div class="overlay_outer"><div class="article-img overlay">';
                    HtmlString += '<img src="/Content/Admin/Documents/CategoryLarge/' + e.LARGE_IMAGE + '" />';
                    HtmlString += '<div class="overlay_cont"><div class="overlay_innercont">';
                    if (e.CATEGORY_TITLE.length < 20) {
                        HtmlString += '<h3><a href="article-list-article-' + e.CATEGORY_ID + '" title="' + e.CATEGORY_TITLE + '" >' + e.CATEGORY_TITLE + '</a></h3>';
                    }
                    else {
                        HtmlString += '<h3><a href="article-list-article-' + e.CATEGORY_ID + '" title="' + e.CATEGORY_TITLE.substring(0, 20) + '" >' + e.CATEGORY_TITLE.substring(0, 20) + '...</a></h3>';
                    }
                    if (e.SUBTITLE.length < 125) {
                        HtmlString += '<p>' + e.SUBTITLE + '</p>';
                    }
                    else {
                        HtmlString += '<p>' + e.SUBTITLE.substring(0, 125) + '...</p>';
                    }
                    HtmlString += '</div></div></div></div>';
                    HtmlString += '</li>';
                });
                $('#HomeCategory').append(HtmlString);

                $(".article-slider li.first-slide").mouseover(function () {
                    $(this).parents('.slider_outerwrap').addClass('move-right');
                });
                $(".article-slider li.first-slide").mouseout(function () {
                    $(this).parents('.slider_outerwrap').removeClass('move-right');
                });

                    $(this).parents('.slider_outerwrap').addClass('move-left');
                });

                    $(this).parents('.slider_outerwrap').removeClass('move-left');
                });

                $('.home_block4 ul li:nth-child(3n)').addClass('first');

                $(".home_block6 .article-img.overlay").mouseover(function () {
                    $(this).addClass('hover');
                });
                $(".home_block6 .article-img.overlay").mouseout(function () {
                    $(this).removeClass('hover');
                });
            }
        });
    }
