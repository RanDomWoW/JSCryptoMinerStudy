                // recursively wrap all words in <span.lb-word>
                (function _wrap_words($root) {
                    $root.contents().each(function() {
                        if (this.nodeType === 3) {
                            var wrapped = $.map(this.nodeValue.split(/\s/), function(word) {
                                return $.trim(word).length > 0

                                       : '';
                            }).join(' ');

                // loop threw <span.lb-word>'s and split parent tags at line breaks
                (function _parse_lines() {
                    var $words = $('span.lb-word', $container),
                        lastTop = $words[0].offsetTop,
                        lastLeft = $words[0].offsetLeft;

                    $words.each(function() {
                        if (this.offsetTop != lastTop && this.offsetLeft <= lastLeft) {
                            if ($(this).parent().children()[0] != this) {
                                split_tag(this, $container); // split_tag() is destructive
                                _parse_lines();      	     // so start over after calling it
                                return false;                // and stop the current $.each loop
                            }
                            lastTop = this.offsetTop;
                        }
                        lastLeft = this.offsetLeft;
                    });
                })();

                function split_tag(node, $root) {
                    var $this = $(node),
                        $parent = $this.parent(),
                        close = '',
                        open = '';

                    $this.attr('class', 'lb-word lb-search');
                    while (true) {
                        close += '</' + $parent[0].tagName + '>';
                        open = $parent[0].outerHTML.substr(
                            0, $parent[0].outerHTML.indexOf('>') + 1
                        ) + open;

                        if ($parent.parent()[0] === $root[0]) break;
                        else $parent = $parent.parent();
                    }

                    var new_element = $parent[0].outerHTML.replace(
                        '<span class="lb-word lb-search">',
                        close + open + '<span class="lb-word">'
                    );
                    $parent.replaceWith(new_element);
                }

	                    $link = $next;
	                    $next = $link.next();
	                }
