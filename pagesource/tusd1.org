    function arrowClick(x) {

        try {
            //counts the position of the arrows
            var clicks = 0;

            //reset all bubble colors
            resetBubbleColors();

            //get the count of all the images we have
            var ct = parseInt(document.getElementById('bubblecount').value) - 1;

            currBubble.value = clicks;
