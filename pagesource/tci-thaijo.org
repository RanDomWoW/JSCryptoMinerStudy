$.fn.clicktoggle = function(a, b) {
    return this.each(function() {
        var clicked = false;
        $(this).click(function() {
            if (clicked) {
                clicked = false;
                return b.apply(this, arguments);
            }
            clicked = true;
            return a.apply(this, arguments);
        });
    });
};

function get_journals(){
  $.get('/public/index/journal_list2.php', function(data) {
    $('#journalsOutput').empty();
          $('#journalsOutput').append('<ul style="display: flex;display: -webkit-flex;-webkit-flex-direction: column;flex-direction: column;flex-wrap: wrap;">');

        $.each(data,function(k,v){
          $('#journalsOutput').append('<li style="color: #FFFFFF;list-style-type: none;margin-left: 10px;"><span class="glyphicon glyphicon-book" aria-hidden="true"> </span> <a style="color: #FFFFFF;" href="/index.php/'+v.path+'/index">'+v.name+'</a></li>');
        });

          $('#journalsOutput').append('</ul>');

        });
}

function get_articles(offset){
  offset = offset ? offset : 0;
  $.get('/public/index/article_list.php?offset='+offset, function(data) {
    $('#articlesOutput').empty();
	  var html = "";
          var nav  =  '<div class="row"><nav class="col-xs-12"><ul class="pager" style="background-color:transparent">';


	if(data['setting'].offset == 0) {
    	   nav += '<li class="previous disabled"><a href="#"><span aria-hidden="true">&larr;</span> Newer</a></li>';
	}else{

    	   nav += '<li class="previous"><a onclick="get_articles('+(parseInt(data['setting'].offset) - parseInt(data['setting'].limit))+')"><span aria-hidden="true">&larr;</span> Newer</a></li>';
	}


        nav += '<li class="next"><a onclick="get_articles('+(parseInt(data['setting'].offset) + parseInt(data['setting'].limit))+')">';

        nav += 'Older <span aria-hidden="true">&rarr;</span></a></li></ul></nav></div>';

	html += nav;
//        html += '<ul style="display: flex;display: -webkit-flex;-webkit-flex-direction: column;flex-direction: column;flex-wrap: wrap;">';
        html += '<ul style="padding-left: 5px;">';

        $.each(data['data'],function(k,v){
           html += '<li style="color: #FFFFFF;list-style-type: none;margin-left: 0px;"><span class="glyphicon glyphicon-book" aria-hidden="true"> </span> <a style="color: #FFFFFF;" href="/index.php/'+v.path+'/article/view/'+v.submission_id+'">'+(v.title ? v.title:'No Title')+'</a></li>';
        });

	html += nav;
        $('#articlesOutput').append(html);

        });
}

$( "a#journalsList" ).clicktoggle(function() {
   $("#journalsGrid").show();
   $("#articlesGrid").hide();
   $("#announcementGrid").hide();
   get_journals();

}, function() {
   $("#journalsGrid").hide();
   $("#announcementGrid").show();
});

$( "a#articlesList" ).clicktoggle(function() {
   $("#articlesGrid").show();
   $("#journalsGrid").hide();
   $("#announcementGrid").hide();
   get_articles();

}, function() {
   $("#articlesGrid").hide();
   $("#announcementGrid").show();

});
