					$.each(urlExemptList,function(i,item){
						if(href.indexOf(item) > -1){
							isExempt = true;
							return false;
						} 
					});
					return isExempt;
				}

					var haveJscript = href.indexOf('javascript') == -1;
		
				   if(!(comp.test(href) || poundSign || checkExemptList(href)) || !haveJscript){

				   }
				   
				});
