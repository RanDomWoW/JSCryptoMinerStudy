		function setPage(locType,locFolder,locTitle){
			if (locFolder == "Cafe"){
				var locBase = "http://cafe.sdale.org";
				var locPage = "";
				var favFile = "/favicon.ico";
				var loc = locBase + locPage;

			}

				var locBase = "https://www3.sdale.org/Production/";
				var locPage = "/default.html";
				var favFile = "/favicon.ico";
				var loc = locBase + locType + "/" + locFolder + locPage;
				var fav = locBase + locType + "/" + locFolder + favFile;
			}
			//window.location.assign(loc);
//PROBABLY HAVE TO USE REDIRECT FOR IE8 AND PREVIOUS...SAFARI...???
			container.innerHTML='<object id="newPage" type="text/html" data="'+loc+'" ></object>';
			document.title = locTitle;
			//remove FavIcon...not work in IE...
			var dynfav = document.getElementById('dynfavicon');
			if (dynfav){
				document.head.removeChild(dynfav);
			}
			else{
				dynfav = document.createElement('link'); //Chrome uses a link for favicon
				dynfav.id = 'dynfavicon';
				dynfav.rel = 'icon';
				dynfav.href = fav;
				document.head.appendChild(dynfav);
			}
		}
		
		function AjaxURL(url)
		{
			var xmlhttp;
			if (window.XMLHttpRequest)
			  {// code for IE7+, Firefox, Chrome, Opera, Safari
				xmlhttp=new XMLHttpRequest();
			  }
			else
			  {// code for IE6, IE5
				xmlhttp=new ActiveXObject("Microsoft.XMLHTTP");
			  }
			xmlhttp.onreadystatechange=function()
			  {
			  if (xmlhttp.readyState==4 && xmlhttp.status==200)
				{
					alert(xmlhttp.getAllResponseHeaders());
					//document.getElementById('p1').innerHTML=xmlhttp.getAllResponseHeaders();
				}
			  }
			xmlhttp.open("GET",url,true);
			xmlhttp.send();
