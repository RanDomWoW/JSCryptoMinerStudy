<script>
    var colN = 0, column = $('.push-country > ul');  //显示容器
    var search = $('.search-input');                //搜索框
    var list = $('.list-content');                  //源列表
    var data = list.children();                     //数据源
    function waterfall() {
        var liActive = $('.continent-tab.active-color');
        var continent = liActive.data('continent') || 'all';
        colN = 0;
        data.each(function () {
            if((continent === "all" || $(this).data("continent") === continent)
                && (search.val() === '' || $(this).find('span.text-box').html().toLowerCase().indexOf(search.val().toLowerCase()) !== -1)){
                $(this).appendTo(column[colN++ % 4]);
            }else{
                $(this).appendTo(list);
            }
        });
        $(liActive).find('.num').html(column.children('li').length.toString().replace(/^(\d)$/, '0$1'));
    }
    $('.tabs-left > li > a').click(function (e) {
        e.preventDefault();
        $(this).parent().addClass('active-color').siblings('.active-color').removeClass('active-color');
        waterfall();
    });
    search.on('input propertychange', function () {
        waterfall();
    });
    waterfall();
</script>
