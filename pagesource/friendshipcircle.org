	var winWidth = $( ".mainStage" ).width();
	var ratio = winWidth/winHeight;
	if(ratio<=1.7){
		$( "#bgvid" ).removeClass( "maxWidth" );
		$( "#bgvid" ).addClass( "maxHeight" );
	}else{
		$( "#bgvid" ).addClass( "maxWidth" );
		$( "#bgvid" ).removeClass( "maxHeight" );
	}

	newPosition=currentFrame*winHeight;
	newPosition=-Math.abs(newPosition)+'px';
	$( ".mainSlider" ).animate({ "top": newPosition }, 0, function() {
		moving=false;
	});
});

	$(".mainStage").click(function(){
		var winWidth = $( ".mainStage" ).width();
		if(winWidth<=switchSize){
			$( ".subMenu" ).addClass( "sideSlide" );
		}
	});

	$(".burger").click(function(){
		var winWidth = $( ".mainStage" ).width();
		if(winWidth<=switchSize){
			$( ".subMenu" ).toggleClass( "sideSlide" );
		}
	});

    $(".arrowButton").click(function(e){

		if(currentFrame==totalFrames-1){

			moveSlider(0);
		}else{
			moveSlider('down');
		}

		e.preventDefault();
    });


	var timer;
	function moveSlider(frame){
		var winHeight = $( ".mainStage" ).height();
		var currentPosition = $( ".mainSlider" ).css("top");
		currentPosition = parseInt(currentPosition, 10);
		var winWidth = $( ".mainStage" ).width();


		if(frame=='up'){
			if(currentFrame!=0){
				moving=true;
				currentFrame-=1;
			}
		}else if(frame=='down'){
			if(currentFrame!=totalFrames-1){
				moving=true;
				currentFrame+=1;
			}
		}else{
			currentFrame=frame;
		}

		if(winWidth>=switchSize){
			if(currentFrame==totalFrames-1){
				//$(".footMenu").hide();
				$( ".footMenu" ).slideUp( "slow" );
			}else{
				//$(".footMenu").show();
				$( ".footMenu" ).slideDown( "slow" );
			}
			//$( ".subMenu" ).removeClass( "sideSlide" );
		}else{
			$( ".subMenu" ).addClass( "sideSlide" );
			$(".footMenu").hide();
		}
		if(currentFrame==8){
			$( ".arrowButton img" ).addClass( "rotated" );
		}else{
			$( ".arrowButton img" ).removeClass( "rotated" );
		}
		console.log(currentFrame);
		//console.log('scrollDown');
		//console.log(currentPosition);
		//console.log(winHeight);
		newPosition=currentFrame*winHeight;
		newPosition=-Math.abs(newPosition)+'px';
		$( ".mainSlider" ).css("top", newPosition);
		/*
		$( ".mainSlider" ).animate({ "top": newPosition }, aniDuration, function() {
			moving=false;
		});
		*/


		clearTimeout(timer);

		timer = setTimeout(function(){
			moving=false;
		}, 500);


	}

});
