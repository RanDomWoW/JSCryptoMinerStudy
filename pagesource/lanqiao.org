			$(function() {
				
				var videoEl1 = $("#video1")[0],
					videoEl2 = $("#video2")[0];
				
				function isPlay(videoElement){
					return videoElement.currentTime > 0 && !videoElement.paused && !videoElement.ended && videoElement.readyState > 2;
				}
				
				
				$('#fullPage').fullpage({
					navigation: true, //显示导航
					navigationTooltips: ['蓝桥招聘','首页',  '蓝桥软件学院', '蓝桥杯大赛', '蓝保社保', '蓝师汇', '有为', '大学创客', '新闻', 'footer'], //导航tips
					anchors: ['zhaopin','home', 'xueyuan', 'dasai', 'lanbao', 'lanshihui', 'youwei', 'chuangke', 'news', 'footer'], //绑定菜单:a标签的href属性
					afterLoad: function(anchorLink, index) { //滚动前的函数
						if(index === 1) {
							$(".header").removeClass("header-bg");
						} else {
							$(".header").addClass("header-bg");
						}
						
						if(index === 3){
							if(!isPlay(videoEl1)){
								videoEl1.play();
							}
						}else{
							if(isPlay(videoEl1)){
								videoEl1.pause();
							}
						}
						
						if(index === 4){
							if(!isPlay(videoEl1)){
								videoEl2.play();
							}
						}else{
							if(isPlay(videoEl1)){
								videoEl2.pause();
							}
						}
						
					},
					afterRender: function() {
						//$("#video1").get(0).play();
						//$("#video2").get(0).play();
					}
				});

				/*for page1 arrow icon*/
				$("#arrow_down").click(function() {
					$.fn.fullpage.moveSectionDown();
				});

				/* 视频播放相关 */
				$playControls = $('.video-player .controls'),
					$mainVideo = $("#mainVideo")[0];

				$("#playBtn").click(function() {
					$('.video-player').fadeIn('fast');
					$playControls.removeClass("play")
					$mainVideo.play()
				})

				$playControls.click(function() {
					if($playControls.hasClass("play")) {
						$mainVideo.play();
					} else {
						$mainVideo.pause();
					}
					$playControls.toggleClass('play')
				})

				$(".video-player .close").click(function() {
					$('.video-player').fadeOut('fast');
					$playControls.addClass("play")
					$mainVideo.pause();
				})
				
				
				//判断是否手机显示(暂时)
				var ua = navigator.userAgent;
				var ipad = ua.match(/(iPad).*OS\s([\d_]+)/),
				isIphone =!ipad && ua.match(/(iPhone\sOS)\s([\d_]+)/),
				isAndroid = ua.match(/(Android)\s+([\d.]+)/),
				isMobile = isIphone || isAndroid;
				//判断			
				if(isMobile){				
					$("html,body").addClass("bhide");
					$(".container").addClass("bhide");
					$(".main").addClass("mright");				
					$(".side").hide();
					$("#service").prop("onclick","");
					$(".video-player .close").addClass("phone");
					$(".section.news .list").addClass("blist");
				}

			});

			/*function for toggle silde*/
			function showSide() {
				$(".container").addClass("go-left-side");
				$("body").find("#fp-nav").hide();
				$(".header").hide();
			}

			function closeSide() {
				$(".container").removeClass("go-left-side");
				$("body").find("#fp-nav").show();
				$(".header").show();
			}
			
			
			
		</script>
	</body>

</html>
