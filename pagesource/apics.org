    $(document).ready(function () {
        GetDefaultEvents();
        if ($("#lo-distance *").length) {
            $("#lo-distance *").attr("disabled", "true");
        }
        TryGeolocation();

    function changeLocation(userLocation) {
        if (document.getElementById("user-location").value || userLocation) {
            var location = document.getElementById("user-location").value;
            if (userLocation) {
                location = userLocation;
            }
            $.ajax({
                url: '/Sitefinity/Public/Services/Custom/LearningOpportunitiesEvents.svc/UpdateUserLocation/' + location,
                cache:false,
                success: function (data) {
                    if (data.Lat.length && data.Lng.length) {
                        getEventsByLocation(data.Lat, data.Lng);
                    }
                    else {
                        document.getElementById("user-location").placeholder = "No location has been found";
                        document.getElementById("user-location").value = '';
                        GetDefaultEvents();
                    }
                }
            });
        }
    }

    function GetDefaultEvents() {
        var Lat = "null";
        var Lng = "null";
        $.ajax({
            url: '/Sitefinity/Public/Services/Custom/LearningOpportunitiesEvents.svc/GetUserCloserEvents/' + Lat + '/' + Lng,
            cache: false,
            success: function (data) {
                drawEvents(data);
            }
        });
    }

    function getUserLocation() {
        $.ajax({
            url: '/Sitefinity/Public/Services/Custom/LearningOpportunitiesEvents.svc/GetUserLocation',
            cache: false,
            success: function (data) {
                if (data) {
                    document.getElementById("user-location").placeholder = data;
                    document.getElementById("user-location").value = '';
                    document.getElementById("lo-distance").style.display = 'block';
                }
            }
        });
    }

    function TryGeolocation() {
        if (navigator.geolocation) {
            navigator.geolocation.getCurrentPosition(SubmitWithGeoData);
        } else {
           getUserLocation();
        }
    }

    function SubmitWithGeoData(position) {
        if ($("#lo-distance").length){
            $("#lo-distance *").removeAttr("disabled");
        }

        var Lat = position.coords.latitude;
        var Lng = position.coords.longitude;
        
        getEventsByLocation(Lat, Lng);
    }

    function getEventsByLocation(Lat, Lng) {
        Lat = Lat.toString().replace(",", ".");
        Lng = Lng.toString().replace(",", ".");

        $.ajax({
            url: '/Sitefinity/Public/Services/Custom/LearningOpportunitiesEvents.svc/GetUserCloserEvents/' + Lat + '/' + Lng,
            cache: false,
            success: function (data) {
                drawEvents(data);
                $.ajax({
                    url: '/Sitefinity/Public/Services/Custom/LearningOpportunitiesEvents.svc/GetUserLocation',
                    cache: false,
                    success: function (data) {
                        if (data) {
                            document.getElementById("user-location").placeholder = data;
                            document.getElementById("user-location").value = '';
                            document.getElementById("lo-distance").style.display = 'block';
                        } else {
                            updateLocation(Lat, Lng);
                            getUserLocation();
                        }
                    }
                });
            }
        });
    }

    function updateGeoLocation(Lat, Lng) {
        $.ajax({
            url: '/Sitefinity/Public/Services/Custom/LearningOpportunitiesEvents.svc/UpdateUserLocation/' + Lat + "," + Lng,
            cache: false,
            success: function (data) {
                if (!data.Lat.length && !data.Lng.length) {
                    document.getElementById("user-location").placeholder = "No location has been found";
                    document.getElementById("user-location").value = '';
                }
            }
        });
    }

    function drawEvents(data) {
        var resultObjects = $.parseJSON(data);
        if (resultObjects.length > 0) {
            $("#user-closer-events li").remove();
            $.each(resultObjects, function (i, item) {
                var title = item.Title;
                if (title.length > 100) {
                    title = title.substring(0, 100) + "...";
                }

                $("#user-closer-events").append('<li class="item"><h3 class="title">' + title +
                                       '<span><a href="' + item.Url + '" class="more">' +
                                       item.Date + '</a></span></h3></li>');
            })
        }
    }

    function goToLO(a) {
        var miles = parseInt($(a).text().match(/\d+/));
        window.location.href = '/learning-opportunities#distance=' + miles;
    }
