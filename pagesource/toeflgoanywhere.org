    expirationDate.setHours(expirationDate.getHours() + 24);
    localStore.set('popupadExpires', expirationDate);
  }

    localStore.set('nopopupAd', 'true');
  }


  if(localStore.get('popupadExpires')) {
    lastPopup = new Date(localStore.get('popupadExpires'));
  }

  if(localStore.get('nopopupAd')) {
    neverShowAgain = true;
  }
  if(!neverShowAgain) {
    if(!localStore.get('popupadExpires') || lastPopup < today) {
      openInterstitialModal();
    }
  }
