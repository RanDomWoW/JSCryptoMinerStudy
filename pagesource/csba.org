    function GetAge(postDate) {
        var retVal = "";
        var oneDay = 86400000;
        var oneHour = 3600000;
        var oneMin = 60000;
        var oneSec = 1000;
        var now = null;
        try {
            now = Date.now();
        }
        catch (err) {
            //for ie8 
            now = new Date().valueOf();
        }
        var testDate = new Date(postDate);
        
        if (isNaN(testDate)) {
            try {
                testDate = Date.parse(postDate.replace(/( \+)/, ' UTC$1'))
                //alert('new post date=' + dateP);
            }
            catch (err1) {
            
            }
        }
        var days = Math.floor(Math.abs((now - testDate) / (oneDay)));
        var hours = Math.floor(Math.abs((now - testDate) / (oneHour))) - (days * 24);
        var mins = Math.floor(Math.abs((now - testDate) / (oneMin))) - (days * 24 * 60) - (hours * 60);
        var secs = Math.floor(Math.abs((now - testDate) / (oneSec))) - (days * 24 * 60 * 60) - (hours * 60 * 60) - (mins * 60);
        if (retVal == "" && days > 0) {
            retVal += days + "d";
        }
        if (retVal == "" && hours > 0) {
            retVal += hours + "h";
        }
        if (retVal == "" && mins > 0) {
            retVal += mins + "m";
        }
        if (retVal == "" && secs > 0) {
            retVal += secs + "s";
        }
        
        return retVal;
    }//END: GetAge

                            videoDataExists = false;

                        videoDataExists = false;

                    var contentReduction = 0;

                    SetHtmlForYouTube(contentReduction, videoDataExists, video, jquerySelector);

                    setTimeout(function () {
                        //wait firstTime before starting setInterval
                        //need wait for content to render
                        if (jQuery(jquerySelector).height() > maxContentHeight) {
                            contentReduction += 50;
                            //now continuously call until good

                                SetHtmlForYouTube(contentReduction, videoDataExists, video, jquerySelector);
                                if (jQuery(jquerySelector).height() > maxContentHeight)
                                    contentReduction += 50;
                                else
                                    clearInterval(intervalId);

                            }, 1);
                        }
                    }, 1);

                },
                error: function (XMLHttpRequest, textStatus, errorThrown) { return 0; }
            });
        }
