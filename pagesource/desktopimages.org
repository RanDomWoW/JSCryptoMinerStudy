	$j(document).ready(function() {
		$j("#sendForm").click(function() {
	
			$j("#error").css("display", "block");
			$j("#error").html("&nbsp;");
			
			$j.post("/jq/sendform/", 
			
			{
				email: $j("#email").val(),
				subject: $j("#subject").val(),
				textvalue: $j("#msg").val()
			},
			function( data ) {
				
				var a = $j.parseJSON(data);
				
				$j("#error").html(a.txt+"<br /><br />");
			});
			return false;
		});
	});
</script>
