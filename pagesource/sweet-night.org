	function bookmarks(type, id) {
		var param = {};      
		
		param['type'] = type;
		param['id'] = id;
		
		$.ajax({
			type:'post',
			url:'http://sweet-night.org/engine/ajax/bookmarks.php',//url адрес файла обработчика
			data:{param:JSON.stringify(param)},
			response:'text',
			success: bookmarks_result
		});	
	}

	function bookmarks_result ( data ) {
		var json = JSON.parse( data );
		if (json.status == 'ok') {
			if (json.type == 'bookmarks_auth') {
				alert(eval(json.type));}
			else {document.getElementById('bookmarks').innerHTML = eval(json.type);}
		}
	}
