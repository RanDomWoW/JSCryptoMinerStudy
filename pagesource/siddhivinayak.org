function randommaker(range) {
	rand=Math.floor(range*Math.random())
    return rand

function initsnow() {
	if (ie5 || opera) {
      marginbottom = document.body.clientHeight
      marginright = document.body.clientWidth
   }
   else if (ns6) {

      marginright = window.innerWidth
