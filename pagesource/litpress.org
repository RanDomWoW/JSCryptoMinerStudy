        function GetCookieVal(name)
        {
            name = name + "=";
            var decodedCookie = decodeURIComponent(document.cookie)
            var ca = decodedCookie.split(';');
            var cval = "";

            for (var i = 0; i < ca.length; i++)
            {
                var c = ca[i];

                while (c.charAt(0) == ' ')
                {
                    c = c.substring(1);
                }

                {
                    cval = c.substring(name.length, c.length)
