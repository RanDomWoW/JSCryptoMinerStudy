                var password = "";

                var okToGo = true;
                if ($("#username").val() == "") {
                    okToGo = false;
                } else {
                    username = $("#username").val();
                }

                if ($("#password").val() == "") {
                    okToGo = false;
                } else {
                    password = $("#password").val();
                }

                if (okToGo) {
                    var userInfo = {
                        UserName: username,
                        Password: password
                    };

                    var processed = JSON.stringify(userInfo);

                    $("#loginPopup").html(getLoader());

                    $.ajax({
                        type: 'post',
                        contentType: 'application/json',
                        dataType: 'json',
                        url: "/Profile/Account/LoginAjax",
                        data: processed,
                        success: function (data, textStatus, jqXHR) {
                            if (data.Success) {
                                if ($("#loggedInButton").hasClass("hide"))
                                    $("#loggedInButton").removeClass("hide");
                                if (!$("#logInButton").hasClass("hide"))
                                    $("#loginButton").addClass("hide");

                                $("#loginDrop").foundation('dropdown', 'close', $("#loginDrop"));

                                if (typeof (loginAction) === "function")
                                    loginAction();

                                $("#registerForAccount").hide();
                                $("#fullName").text(data.FullName);

                                if (data.IsAdmin) {
                                    if ($("#adminPanel").hasClass("hidden"))
                                        $("#adminPanel").removeClass("hidden");
                                }

                                //Todo: Do this better.
                                if (data.IsLeader)
                                    location.reload();
                            } else {
                                $("#errorMessageAjax").html(data.Message);
                                $("#loginPopup").html("Log In");
                            }
                        },
                        error: function (jqXHR, textStatus, errorThrown) {
                            $("#errorMessageAjax").html("There was an issue logging in. Check your username and password and try again.");
                        }
                    });
                } else {
                    $("#errorMessageAjax").html("Please enter a username and a password and try again.");
                }
            });

        function drawHighcharts(fflCountVal, sppCountVal, ppCountVal, demoCountVal, epiCountVal, applyCountVal) {
            var fflScore = [];
            var sppScore = [];
            var ppScore = [];
            var demoScore = [];
            var epiScore = [];
            var applyScore = [];

            fflScore.push(fflCountVal);
            sppScore.push(sppCountVal);
            epiScore.push(epiCountVal);
            ppScore.push(ppCountVal);
            demoScore.push(demoCountVal);
            applyScore.push(applyCountVal);

            //var colors = Highcharts.getOptions().colors;
            var colors = ["#e59d83", "#a186be", "#b7d887", "#618bae", "#0b3c58"];
            var categories = ['FFL', 'EPI', 'SPP', 'Topics Of Interest', 'Demographics'];
            var data = [{
                y: 56.33,
                color: colors[0],
                drilldown: {
                    name: 'FFL',
                    categories: ['FFL'],
                    data: fflScore,
                    color: colors[0],
                }
            }, {
                y: 10.38,
                color: colors[1],
                drilldown: {
                    name: 'EPI',
                    categories: ['EPI'],
                    data: epiScore,
                    color: colors[2]
                }
            }, {
                y: 24.03,
                color: colors[2],
                drilldown: {
                    name: 'SPP',
                    categories: ['SPP'],
                    data: sppScore,
                    color: colors[3]
                }
            }, {
                y: 4.77,
                color: colors[3],
                drilldown: {
                    name: 'Topics Of Interest',
                    categories: ['Topics Of Interest'],
                    data: ppScore,
                    color: colors[2]
                }
            }, {
                y: 4.77,
                color: colors[4],
                drilldown: {
                    name: 'Demographics',
                    categories: ['Demographics'],
                    data: demoScore,
                    color: colors[2]
                }
            }];
            var browserData = [];
            var versionsData = [];
            var i;
            var j;
            var dataLen = data.length;
            var drillDataLen;
            var brightness;


            // Build the data arrays
            for (i = 0; i < dataLen; i += 1) {

                // add browser data
                browserData.push({
                    name: categories[i],
                    y: data[i].y,
                    color: data[i].color
                });

                // add version data
                drillDataLen = data[i].drilldown.data.length;
                for (j = 0; j < drillDataLen; j += 1) {
                    brightness = 0.2 - (j / drillDataLen) / 5;
                    versionsData.push({
                        name: data[i].drilldown.categories[j],
                        y: data[i].drilldown.data[j],
                        color: Highcharts.Color(data[i].color).get()
                    });
                }
            }

            var innerDonut = '<span style="font-size: 3em; font-weight: 700; color:#082a3d;" id="donutCount">' + applyCountVal + '</span>';

            // Create the chart
            $('#donut').highcharts({
                chart: {
                    type: 'pie'
                },
                title: {
                    text: innerDonut,
                    align: 'center',
                    verticalAlign: 'middle',
                    y: 3,
                    useHtml: true
                },
                plotOptions: {
                    pie: {
                        shadow: false,
                        center: ['50%', '50%']
                    }
                },
                credits: {
                    enabled: false
                },
                exporting: {
                    enabled: false
                },
                series: [{
                    cursor: 'pointer',
                    name: 'Count',
                    data: versionsData,
                    size: '100%',
                    innerSize: '60%',
                    dataLabels: {
                        formatter: function () {
                        }
                    },
                    point: {
                        events: {
                            click: function (event) {
                                var name = this.name;
                                var url = "";

                                switch (name) {
                                    case "FFL": url = "/Profile/Account/MyProfile"; break;
                                    case "PP": url = "/Profile/Account/MyProfile"; break;
                                    case "DEMO": url = "/Profile/Account/MyProfile"; break;
                                    case "EPI": url = "/LeadershipAnalyze/Analyze/CPTIndex"; break;
                                    case "SPP": url = "/LeadershipAnalyze/Analyze/SPPIndex"; break;
                                }

                                window.location.href = url;
                            }
                        }
                    }
                }]
            });

            $("#donutCount").text(applyCountVal);
        }
