        var _MS_PER_DAY = 1000 * 60 * 60 * 24;

        // a and b are javascript Date objects
        function dateDiffInDays(a, b) {
            // Discard the time and time-zone information.
            var utc1 = Date.UTC(a.getFullYear(), a.getMonth(), a.getDate());
            var utc2 = Date.UTC(b.getFullYear(), b.getMonth(), b.getDate());

            return Math.floor((utc2 - utc1) / _MS_PER_DAY);
        }

        function displayTermsOfService() {
            var sNeedTermsOfServiceNotice = $("#hfNeedTermsOfServiceNotice").val();
            if (sNeedTermsOfServiceNotice !== '') {
                var bNeedTermsOfServiceNotice = new Boolean(sNeedTermsOfServiceNotice);

                if (bNeedTermsOfServiceNotice) {
                    $("#divTOSModal").show();
                    $(".spawn").trigger('click');
                    $('#tosAccept')
                        .click(function (e) {
                            e.preventDefault();
                            handleUserTOSChoice(true);
                        });
                    $('#tosDecline')
                        .click(function (e) {
                            e.preventDefault();
                            handleUserTOSChoice(false);
                        });
                }
            } else {

                displayNeedToUpdateProfileMessage();
            }
        }

        function handleUserTOSChoice(userChoice) {
            var dataString = "{'iUserID':'0', 'bUserAgrees': '" + userChoice + "'}";
            $.ajax({
                type: "POST",
                url: "/webservices/TOSManager.asmx/HandleUserChoice",
                cache: false,
                dataType: "text",
                data: dataString,
                contentType: "application/json; charset=utf-8",
                success: function (e) {
                    // Regardless of the user's choice, refresh the page.
                    location.reload(true);
                }
            });
        }

        function displayNeedToUpdateProfileMessage() {
            var path = window.location.pathname;
            var search = window.location.search;
            var retVal = encodeURIComponent(path + search);
            if (path.lastIndexOf("myprofile") == -1) {
                var sNeedToUpdateProfile = $("#hfNeedToUpdateProfile").val();

                        $(".spawn").trigger('click');
                        $('#mypAccept').click(function (e) {
                            e.preventDefault();
                            location.href = "/myprofile?retVal=" + retVal;
                        });
                    }
                }
            }
        }
